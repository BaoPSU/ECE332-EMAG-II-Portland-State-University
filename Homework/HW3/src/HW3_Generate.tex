\begin{notebox}
\textbf{Symbol reference} (used throughout):
\begin{itemize}\setlength{\itemsep}{0pt}\setlength{\topsep}{1pt}\setlength{\parsep}{0pt}
\item $\varepsilon_0 = 8.854\!\times\!10^{-12}$ F/m --- permittivity of free space (constant)
\item $\mu_0 = 4\pi\!\times\!10^{-7}$ H/m --- permeability of free space (constant)
\item $\varepsilon_r$ --- relative permittivity (dielectric constant) of the medium, dimensionless
\item $\mu_r$ --- relative permeability of the medium, dimensionless. $\mu_r = 1$ for nonmagnetic materials (air, most dielectrics)
\item $\varepsilon = \varepsilon_r \varepsilon_0$ --- absolute permittivity of the medium (F/m)
\item $\mu = \mu_r \mu_0$ --- absolute permeability of the medium (H/m)
\item $\sigma$ --- conductivity of the medium (S/m). $\sigma=0$ for lossless / ideal dielectric
\item $\eta = \sqrt{\mu/\varepsilon}$ --- intrinsic impedance of the medium ($\Omega$)
\item $\eta_0 = \sqrt{\mu_0/\varepsilon_0} = 120\pi \approx 377\,\Omega$ --- intrinsic impedance of free space
\item $k = \omega\sqrt{\mu\varepsilon} = \omega\sqrt{\varepsilon_r}/c$ (nonmag.) --- wavenumber (rad/m)
\item $\alpha$ --- attenuation constant (Np/m); $\beta$ --- phase constant (rad/m)
\item $\Gamma = (\eta_2-\eta_1)/(\eta_2+\eta_1)$ --- reflection coefficient (unitless)
\item $\tau = 1 + \Gamma$ --- transmission coefficient (unitless)
\item Superscripts on $\tilde{\mathbf{E}}$: \textbf{$i$} = incident, \textbf{$r$} = reflected, \textbf{$t$} = transmitted. Three distinct waves, three distinct superscripts.
\end{itemize}
\textit{\color{coral}\textbf{Note:} the official HW3 solution PDF labels both reflected and transmitted as $\tilde{E}^r$ (typo). The second one is really $\tilde{E}^t$ --- you can tell because it carries $\tau$ and $k_2$ instead of $\Gamma$ and $k_1$. This doc uses the correct $i/r/t$ throughout.}
\end{notebox}

\probhead{1 --- LHC Wave Normally Incident on a Dielectric}{20 pts}

\textbf{Setup:} 200 MHz LHC plane wave in air, $|E|=5$ V/m, normally incident on dielectric ($\varepsilon_r=4$) at $z\ge0$. Positive max at $z=0$, $t=0$.

\begin{cssnip}
\includegraphics[width=0.49\linewidth]{cs_table82}\hfill
\includegraphics[width=0.49\linewidth]{cs_phasor_templates}\\[4pt]
\includegraphics[width=0.49\linewidth]{cs_power}
\end{cssnip}

\textbf{Wave parameters in medium 1 (air):}

Angular frequency from $f$:
\[\omega = 2\pi f = 2\pi(2\!\times\!10^8) = 4\pi\!\times\!10^8~\text{rad/s}\]

Wavenumber from the lossless plane wave formula $k = \omega\sqrt{\varepsilon_r}/c$. In air, $\varepsilon_r=1$ so this reduces to $k_1 = \omega/c$:
\[k_1 = \tfrac{\omega}{c} = \tfrac{4\pi\!\times\!10^8}{3\!\times\!10^8} = \tfrac{4\pi}{3}~\text{rad/m}\]

\textbf{(a) LHC incident phasor.} The PHASOR TEMPLATES section says the incident wave has the form $\tilde{\mathbf{E}}^i = a_0\hat{e}\,e^{-jk_1z}$ where $\hat{e}$ depends on polarization. From the polarization table:
\[\text{LHC} \Rightarrow \hat{e} = \hat{x}+j\hat{y}\]

With $a_0 = 5$ V/m (the modulus) and $k_1 = 4\pi/3$:
\[\boxed{\tilde{\mathbf{E}}^i = 5(\hat{x}+j\hat{y})\,e^{-j4\pi z/3}~\text{V/m}}\]

\textbf{(b) Reflection coefficient $\Gamma$ and transmission coefficient $\tau$.}

\textit{General intrinsic impedance:} $\eta = \sqrt{\mu/\varepsilon}$.

\textit{Decompose into relative + free-space constants:} since $\mu = \mu_r\mu_0$ and $\varepsilon = \varepsilon_r\varepsilon_0$,
\[\eta = \sqrt{\tfrac{\mu_r\mu_0}{\varepsilon_r\varepsilon_0}} = \sqrt{\tfrac{\mu_r}{\varepsilon_r}}\cdot\sqrt{\tfrac{\mu_0}{\varepsilon_0}} = \sqrt{\tfrac{\mu_r}{\varepsilon_r}}\,\eta_0\]
which for nonmagnetic media ($\mu_r=1$) reduces to $\eta = \eta_0/\sqrt{\varepsilon_r}$.

\textit{Compute $\eta_0$ from $\mu_0, \varepsilon_0$ directly:}
\[\eta_0 = \sqrt{\tfrac{\mu_0}{\varepsilon_0}} = \sqrt{(4\pi\!\times\!10^{-7})(36\pi\!\times\!10^9)} = 120\pi\,\Omega \approx 377\,\Omega\]

\textit{Medium 1 (air):} $\mu_r=1, \varepsilon_r=1$. So $\eta_1 = \eta_0\sqrt{1/1} = \eta_0 = 120\pi\,\Omega$.

\textit{Medium 2 (nonmagnetic dielectric, $\mu_r=1, \varepsilon_r=4$):} use the shortcut $\eta = \eta_0/\sqrt{\varepsilon_r}$:
\[\eta_2 = \tfrac{\eta_0}{\sqrt{\varepsilon_{r2}}} = \tfrac{120\pi}{\sqrt{4}} = \tfrac{120\pi}{2} = 60\pi\,\Omega\]

\textit{Now apply the boundary formula} $\Gamma = (\eta_2-\eta_1)/(\eta_2+\eta_1)$:
\[\Gamma = \tfrac{60\pi - 120\pi}{60\pi + 120\pi} = \tfrac{-60\pi}{180\pi} = \boxed{-\tfrac{1}{3}}\]

\[\tau = 1+\Gamma = 1 - \tfrac{1}{3} = \boxed{\tfrac{2}{3}}\]

\textbf{(c) Reflected, transmitted, and total field phasors.}

\textit{Wavenumbers in both media} use the same formula $k_i = \omega\sqrt{\varepsilon_{r,i}}/c$, just plug in each medium's $\varepsilon_r$:

\textit{Medium 1 (air, $\varepsilon_{r,1}=1$):}
\[k_1 = \tfrac{\omega\sqrt{1}}{c} = \tfrac{4\pi\!\times\!10^8}{3\!\times\!10^8} = \tfrac{4\pi}{3}~\text{rad/m}\]

\textit{Medium 2 (dielectric, $\varepsilon_{r,2}=4$):}
\[k_2 = \tfrac{\omega\sqrt{4}}{c} = \tfrac{2\omega}{c} = 2k_1 = \tfrac{8\pi}{3}~\text{rad/m}\]

\textit{Notice}: same formula, only $\varepsilon_r$ changes per medium (just like $\eta$).

\textit{From the PHASOR TEMPLATES} (reflected wave travels in $-\hat{z}$, sign of $z$-term flips):
\[\tilde{\mathbf{E}}^r = \Gamma\cdot a_0\hat{e}\,e^{+jk_1z} = -\tfrac{1}{3}\!\cdot\!5(\hat{x}+j\hat{y})\,e^{+j4\pi z/3} = -\tfrac{5}{3}(\hat{x}+j\hat{y})\,e^{+j4\pi z/3}\]

\textit{Transmitted wave in lossless medium 2:}
\[\tilde{\mathbf{E}}^t = \tau\cdot a_0\hat{e}\,e^{-jk_2z} = \tfrac{2}{3}\!\cdot\!5(\hat{x}+j\hat{y})\,e^{-j8\pi z/3} = \tfrac{10}{3}(\hat{x}+j\hat{y})\,e^{-j8\pi z/3}\]

\textit{Total field in medium 1 is incident + reflected:}
\[\boxed{\tilde{\mathbf{E}}_1 = \tilde{\mathbf{E}}^i + \tilde{\mathbf{E}}^r = 5(\hat{x}+j\hat{y})\!\left(e^{-j4\pi z/3} - \tfrac{1}{3}e^{+j4\pi z/3}\right)~\text{V/m}}\]

\textbf{(d) Percentages of incident average power reflected and transmitted.}

\textit{From the POWER REFLECTION snippet,} $\%P_r = 100|\Gamma|^2$ and $\%P_t = 100|\tau|^2(\eta_1/\eta_2)$. The $\eta_1/\eta_2$ factor accounts for the change in wave impedance: even though $|\tau|>0$, the medium beyond carries power differently.

\[\%P_r = 100\!\cdot\!\left|-\tfrac{1}{3}\right|^2 = 100\!\cdot\!\tfrac{1}{9} = \boxed{11.1\%}\]
\[\%P_t = 100\!\cdot\!\left|\tfrac{2}{3}\right|^2\!\cdot\!\tfrac{120\pi}{60\pi} = 100\!\cdot\!\tfrac{4}{9}\!\cdot\!2 = \boxed{88.9\%}\]

\textit{Sanity check:} $11.1\% + 88.9\% = 100\%$. Energy conserved.

\probhead{2 --- Same Wave, Poor-Conductor Medium 2}{20 pts}

\textbf{Setup:} Replace medium 2 with poor conductor: $\varepsilon_r=2.25$, $\mu_r=1$, $\sigma=10^{-4}$ S/m. Everything else same as P1.

\begin{cssnip}
\includegraphics[width=0.49\linewidth]{cs_table82}\hfill
\includegraphics[width=0.49\linewidth]{cs_phasor_templates}\\[4pt]
\includegraphics[width=0.49\linewidth]{cs_lowloss}\hfill
\includegraphics[width=0.49\linewidth]{cs_power}
\end{cssnip}

\textbf{Wave parameters in medium 1 (air).} Same as P1, since medium 1 hasn't changed:
\[\omega = 2\pi f = 4\pi\!\times\!10^8~\text{rad/s},\quad k_1 = \tfrac{\omega\sqrt{\varepsilon_{r,1}}}{c} = \tfrac{\omega}{c} = \tfrac{4\pi}{3}~\text{rad/m},\quad \eta_1 = \eta_0 = 120\pi\,\Omega\]

\textbf{(a) LHC incident phasor.} Identical to P1 â€” incident wave is in medium 1 (air), which didn't change. From the PHASOR TEMPLATES + polarization table ($\hat{e}_{\text{LHC}} = \hat{x}+j\hat{y}$):
\[\boxed{\tilde{\mathbf{E}}^i = 5(\hat{x}+j\hat{y})\,e^{-j4\pi z/3}~\text{V/m}}\]

\textbf{(b) Reflection and transmission coefficients --- but first classify medium 2.}

\textit{Compute the loss tangent $\sigma/(\omega\varepsilon)$ to pick which formulas to use:}
\[\tfrac{\sigma_2}{\omega\varepsilon_2} = \tfrac{\sigma_2}{\omega\varepsilon_{r,2}\varepsilon_0} = \tfrac{10^{-4}}{(4\pi\!\times\!10^8)(2.25)(8.854\!\times\!10^{-12})} = 4\!\times\!10^{-3}\]
This is $\ll 1$, so medium 2 is a \textbf{low-loss dielectric}. The LOSSY MEDIA --- 5 CASES table classifies media by $\sigma/(\omega\varepsilon)$ and tells us to use the \emph{low-loss row} shortcuts (not the full exact lossy formulas in the bottom).

\textit{Intrinsic impedance --- same derivation chain as P1, with the low-loss correction.} The general formula $\eta = \sqrt{\mu_r/\varepsilon_r}\,\eta_0$ lives in the intrinsic-impedance footer of TABLE 8-2. For low-loss media the low-loss row of LOSSY MEDIA --- 5 CASES says:
\[\eta_2 \approx \sqrt{\tfrac{\mu}{\varepsilon'}}\!\left(1 + j\tfrac{\sigma}{2\omega\varepsilon'}\right)\quad\xrightarrow[\sigma/\omega\varepsilon\ll 1]{\mu_r=1}\quad \tfrac{\eta_0}{\sqrt{\varepsilon_r}}\]
The $j$ correction is $4\!\times\!10^{-3}/2 = 2\!\times\!10^{-3}$, tiny enough to drop for $\Gamma, \tau$ computation. So:
\[\eta_2 \approx \tfrac{\eta_0}{\sqrt{\varepsilon_{r,2}}} = \tfrac{120\pi}{\sqrt{2.25}} = \tfrac{120\pi}{1.5} = 80\pi\,\Omega\]

\textit{Attenuation and phase constants from low-loss row of LOSSY MEDIA --- 5 CASES} (these matter for the transmitted-wave decay even though they don't affect $\Gamma$):
\[\alpha_2 \approx \tfrac{\sigma_2 \eta_0}{2\sqrt{\varepsilon_{r,2}}} = \tfrac{10^{-4}\cdot 120\pi}{2\sqrt{2.25}} = 1.26\!\times\!10^{-2}~\text{Np/m}\]
\[\beta_2 \approx \tfrac{\omega\sqrt{\varepsilon_{r,2}}}{c} = \tfrac{(4\pi\!\times\!10^8)(1.5)}{3\!\times\!10^8} = 2\pi~\text{rad/m}\]

\textit{Apply the boundary formulas:}
\[\Gamma = \tfrac{\eta_2 - \eta_1}{\eta_2 + \eta_1} = \tfrac{80\pi - 120\pi}{80\pi + 120\pi} = \tfrac{-40\pi}{200\pi} = \boxed{-\tfrac{1}{5}}\]
\[\tau = 1+\Gamma = 1 - \tfrac{1}{5} = \boxed{\tfrac{4}{5}}\]

\textbf{(c) Reflected, transmitted, and total field phasors.}

\textit{Direction signs from the DIRECTION TABLE.} Med 2 is at $z\ge 0$ (same as P1), so use the first row: incident gets $-jk_1z$, reflected gets $+jk_1z$, transmitted gets the lossy decay $e^{-\alpha_2z}e^{-j\beta_2z}$ (since med 2 is now lossy, replace $e^{-jk_2z}$ with $e^{-\alpha_2z}e^{-j\beta_2z}$).

\textit{Reflected template (still in lossless medium 1, traveling $-\hat{z}$):}
\[\tilde{\mathbf{E}}^r = \Gamma\,a_0\hat{e}\,e^{+jk_1z}\]
Plug in $\Gamma=-1/5$, $a_0=5$, $\hat{e}=\hat{x}+j\hat{y}$, $k_1=4\pi/3$:
\[\tilde{\mathbf{E}}^r = -\tfrac{1}{5}\cdot 5(\hat{x}+j\hat{y})\,e^{+j4\pi z/3} = -(\hat{x}+j\hat{y})\,e^{+j4\pi z/3}~\text{V/m}\]

\textit{Transmitted template (lossy medium 2, traveling $+\hat{z}$, replace $e^{-jk_2z}$ with $e^{-\alpha_2z}e^{-j\beta_2z}$):}
\[\tilde{\mathbf{E}}^t = \tau\,a_0\hat{e}\,e^{-\alpha_2 z}\,e^{-j\beta_2 z}\]
Plug in $\tau=4/5$, $a_0=5$, $\hat{e}=\hat{x}+j\hat{y}$, $\alpha_2=1.26\!\times\!10^{-2}$, $\beta_2=2\pi$:
\[\tilde{\mathbf{E}}^t = \tfrac{4}{5}\cdot 5(\hat{x}+j\hat{y})\,e^{-\alpha_2 z}e^{-j\beta_2 z} = 4(\hat{x}+j\hat{y})\,e^{-1.26\!\times\!10^{-2}z}\,e^{-j2\pi z}~\text{V/m}\]

\textit{Total field in medium 1 ($z\le 0$):}
\[\boxed{\tilde{\mathbf{E}}_1 = \tilde{\mathbf{E}}^i + \tilde{\mathbf{E}}^r = 5(\hat{x}+j\hat{y})\!\left(e^{-j4\pi z/3} - \tfrac{1}{5}e^{+j4\pi z/3}\right)~\text{V/m}}\]

\textbf{(d) Percentages of incident average power reflected and transmitted.}

\textit{From POWER REFLECTION snippet,} same formulas as P1:
\[\%P_r = 100|\Gamma|^2 = 100\!\cdot\!\left|-\tfrac{1}{5}\right|^2 = 100\!\cdot\!\tfrac{1}{25} = \boxed{4\%}\]
\[\%P_t = 100|\tau|^2\,\tfrac{\eta_1}{\eta_2} = 100\!\cdot\!\left|\tfrac{4}{5}\right|^2\!\cdot\!\tfrac{120\pi}{80\pi} = 100\!\cdot\!\tfrac{16}{25}\!\cdot\!\tfrac{3}{2} = \boxed{96\%}\]

\textit{Sanity check:} $4\% + 96\% = 100\%$. Less reflection than P1 (4\% vs 11.1\%) because medium 2 is closer in impedance to air ($\eta_2 = 80\pi$ vs $60\pi$); the smaller impedance jump means more of the wave gets through.

\probhead{3 --- Parallel-Polarized Wave Through Two Dielectric Layers}{10 pts}

\textbf{Setup:} $\theta_i=30^\circ$ from air, layer 1 ($\varepsilon_r=6.25$, $t=5$ cm), layer 2 ($\varepsilon_r=2.25$, $t=5$ cm), exits to air.

\begin{cssnip}
\includegraphics[width=0.49\linewidth]{cs_snell}\hfill
\includegraphics[width=0.49\linewidth]{cs_lateral}\\[4pt]
\includegraphics[width=0.95\linewidth]{cs_table82}
\end{cssnip}

\textbf{(a) Chained Snell's law:}
\[\sin\theta_2 = \sin 30^\circ\sqrt{\tfrac{1}{6.25}} = 0.5\!\cdot\!0.4 = 0.2 \Rightarrow \theta_2 = \sin^{-1}(0.2) = \boxed{11.54^\circ}\]
\[\sin\theta_3 = 0.2\sqrt{\tfrac{6.25}{2.25}} = 0.2\!\cdot\!\tfrac{5}{3} = \tfrac{1}{3} \Rightarrow \theta_3 = \sin^{-1}(1/3) = \boxed{19.47^\circ}\]
\[\sin\theta_4 = \tfrac{1}{3}\sqrt{\tfrac{2.25}{1}} = \tfrac{1}{3}\!\cdot\!\tfrac{3}{2} = \tfrac{1}{2} \Rightarrow \theta_4 = \boxed{30^\circ = \theta_i}\]

\textbf{(b) Lateral displacement:}
\[d = t_1\tan\theta_2 + t_2\tan\theta_3 = 5\tan 11.54^\circ + 5\tan 19.47^\circ = 5(0.2041 + 0.3536)~\text{cm}\]
\[\boxed{d = 2.79~\text{cm}}\]

\probhead{4 --- Dielectric Waveguide TE Mode Identification}{20 pts}

\textbf{Setup:} TE wave, $a=5$ cm, $b=3$ cm, $E_x = -36\cos(40\pi x)\sin(100\pi y)\sin\!\big((2.4\pi\!\times\!10^{10})t - 52.9\pi z\big)$ V/m.

\begin{cssnip}
\includegraphics[width=0.49\linewidth]{cs_wgmodes}\hfill
\includegraphics[width=0.49\linewidth]{cs_table83}\\[4pt]
\includegraphics[width=0.49\linewidth]{cs_epsilon_disp}\hfill
\includegraphics[width=0.49\linewidth]{cs_cutoff}\\[4pt]
\includegraphics[width=0.49\linewidth]{cs_wgimpedance}
\end{cssnip}

\textbf{(a) Mode from the field expression.} Read $m\pi/a$ from the $\cos$ coefficient and $n\pi/b$ from the $\sin$ coefficient.
\[\tfrac{m\pi}{a} = 40\pi \Rightarrow m = 40\!\cdot\!0.05 = 2\]
\[\tfrac{n\pi}{b} = 100\pi \Rightarrow n = 100\!\cdot\!0.03 = 3\]
\[\boxed{\text{Mode: TE}_{23}}\]

\textbf{(b) $\varepsilon_r$ from dispersion}, with $\omega = 2.4\pi\!\times\!10^{10}$ rad/s, $\beta = 52.9\pi$ rad/m:
\[\varepsilon_r = \tfrac{c^2}{\omega^2}\!\left[\beta^2 + \!\left(\tfrac{m\pi}{a}\right)^{\!2}\!+\!\left(\tfrac{n\pi}{b}\right)^{\!2}\right]
  = \!\left(\tfrac{3\!\times\!10^8}{2.4\pi\!\times\!10^{10}}\right)^{\!2}\!\big[(52.9\pi)^2 + (40\pi)^2 + (100\pi)^2\big]\]
\[\boxed{\varepsilon_r = 2.25}\]

\textbf{(c) Cutoff frequency for TE$_{23}$:}
\[u_{p0} = \tfrac{c}{\sqrt{\varepsilon_r}} = \tfrac{3\!\times\!10^8}{1.5} = 2\!\times\!10^8~\text{m/s}\]
\[f_{23} = \tfrac{u_{p0}}{2}\sqrt{\!\left(\tfrac{2}{a}\right)^{\!2}\!+\!\left(\tfrac{3}{b}\right)^{\!2}}
  = 10^8\sqrt{(2/0.05)^2 + (3/0.03)^2} = 10^8\sqrt{1600+10000}\]
\[\boxed{f_{23} = 10.77~\text{GHz}}\]

\textbf{(d) $H_y$ from $Z_{\text{TE}}$:} operating frequency $f = \omega/(2\pi) = 12$ GHz.
\[Z_{\text{TE}} = \tfrac{\eta}{\sqrt{1-(f_c/f)^2}} = \tfrac{377/\sqrt{2.25}}{\sqrt{1-(10.77/12)^2}}
  = \tfrac{251.3}{\sqrt{1-0.805}} = \tfrac{251.3}{0.442} = 569.9~\Omega\]
\[\boxed{H_y = \tfrac{E_x}{Z_{\text{TE}}} = -0.063\cos(40\pi x)\sin(100\pi y)\sin((2.4\pi\!\times\!10^{10})t - 52.9\pi z)~\text{A/m}}\]
