\shead{amberbg}{amber}{MAGNETIC FLUX \& FARADAY'S LAW}
\begin{ebox}
\eq{Magnetic flux}{Wb = V$\cdot$s}
\[\Phi = \iint_S \mathbf{B}\cdot d\mathbf{s}\]
\eq{Faraday's law}{$N$: number of turns ($N\!=\!1$ for single loop)}
\[V_\text{emf} = -N\frac{d\Phi}{dt}\;\text{(V)}\]
\textit{Lenz's law: induced current direction opposes the change in $\Phi$.}
\end{ebox}

\vspace{0.3pt}
\shead{amberbg}{amber}{TRANSFORMER \& MOTIONAL EMF}
\begin{ebox}
\eq{Transformer EMF}{stationary loop, time-varying $\mathbf{B}$}
\[V_\text{emf}^\text{tr} = -N\iint_S\frac{\partial\mathbf{B}}{\partial t}\cdot d\mathbf{s}\]
\eq{Motional EMF}{moving conductor in $\mathbf{B}$}
\[V_\text{emf}^\text{m} = \oint_C (\mathbf{u}\times\mathbf{B})\cdot d\mathbf{l}\]
\eq{Total EMF}{general case}
\[V_\text{emf} = V_\text{emf}^\text{tr} + V_\text{emf}^\text{m}\]
\textit{$\mathbf{u}$: conductor velocity (m/s).\enspace Choose contour direction by RHR.}
\end{ebox}

\vspace{0.3pt}
\shead{amberbg}{amber}{AC GENERATOR}
\begin{ebox}
\[V_\text{emf} = NA\omega B_0\sin(\omega t + C_0)\;\text{(V)}\]
\textit{$N$: turns;\enspace$A$: loop area (m$^2$);\enspace$B_0$: field amp (T);\enspace$\omega$: rotation (rad/s).}\\
\textit{Derived: $\Phi=NAB_0\cos(\omega t\!+\!C_0)$, $V_\text{emf}=-d\Phi/dt$.\enspace Peak $V_0=NA\omega B_0$.}
\end{ebox}

\vspace{0.3pt}
\shead{tealbg}{teal}{IDEAL TRANSFORMER}
\begin{ebox}
\[\frac{V_1}{V_2} = \frac{N_1}{N_2},\qquad \frac{I_1}{I_2} = \frac{N_2}{N_1}\]
\[Z_\text{in}=\left(\frac{N_1}{N_2}\right)^{\!2}Z_L,\qquad R_\text{in}=\left(\frac{N_1}{N_2}\right)^{\!2}R_L\]
\textit{Lossless: $V_1I_1=V_2I_2$ (power conserved).}\\
\textit{Step-up ($N_2>N_1$): $V_2>V_1$, $I_2<I_1$;\enspace step-down: opposite.}
\end{ebox}

\vspace{0.3pt}
\shead{bluebg}{blue}{DISPLACEMENT CURRENT}
\begin{ebox}
\[\mathbf{J}_d = \frac{\partial\mathbf{D}}{\partial t}=\varepsilon\frac{\partial\mathbf{E}}{\partial t}\;\text{(A/m}^2\text{)},\qquad I_d=\iint_S\mathbf{J}_d\cdot d\mathbf{s}\;\text{(A)}\]
\textit{Phasor: $\tilde{\mathbf{J}}_d=j\omega\varepsilon\tilde{\mathbf{E}}$.\enspace Completes Amp\`ere; enables EM waves in free space.}
\end{ebox}

\vspace{0.3pt}
\shead{coralbg}{coral}{CHARGE CONTINUITY \& RELAXATION}
\begin{ebox}
\[\nabla\cdot\mathbf{J}=-\frac{\partial\rho_v}{\partial t}\;\Longrightarrow\;\oint_S\mathbf{J}\cdot d\mathbf{s}=0\;\text{(Kirchhoff's current law)}\]
\[\tau_r=\frac{\varepsilon}{\sigma}\;\text{(s)},\qquad\rho_v(t)=\rho_{v0}\,e^{-t/\tau_r}\;\text{(C/m}^3\text{)}\]
\textit{Good conductor: $\tau_r\!\approx\!10^{-19}$\,s;\enspace good insulator: $\tau_r\!\approx\!10^{4}$\,s.}
\end{ebox}

\vspace{0.3pt}
\shead{bluebg}{blue}{DYNAMIC (EM) POTENTIALS}
\begin{ebox}
\[\mathbf{B}=\nabla\times\mathbf{A},\qquad\mathbf{E}=-\nabla V-\frac{\partial\mathbf{A}}{\partial t}\]
\textit{$V$: scalar electric potential (V);\enspace$\mathbf{A}$: vector magnetic potential (Wb/m).}\\
\textit{$\mathbf{E}=-\nabla V$ holds only in statics;\enspace $\nabla\times\mathbf{A}=\mathbf{B}$ valid in all cases.}
\end{ebox}

%────── COL 3: POWER + ATTENUATION + SURFACE ──────
\shead{greenbg}{green}{LOSSY MEDIUM POYNTING}
\begin{ebox}
\eq{Average (general)}{W/m$^2$}
\[\mathbf{S}_\text{av}=\tfrac{1}{2}\operatorname{Re}\{\tilde{\mathbf{E}}\times\tilde{\mathbf{H}}^*\}\]
\eq{Lossy medium}{$\eta_c=|\eta_c|e^{j\theta_\eta}$,\;wave in $+\hat{z}$}
\[\mathbf{S}_\text{av}=\frac{|\tilde{E}(0)|^2}{2|\eta_c|}\,e^{-2\alpha z}\cos\theta_\eta\;\hat{z}\quad\text{(W/m}^2\text{)}\]
\eq{Lossless}{$\alpha=0$,\;$\theta_\eta=0$,\;$|\eta_c|=\eta$}
\[\mathbf{S}_\text{av}=\frac{|E_0|^2}{2\eta}\,\hat{k}\]
\textit{Power $\propto e^{-2\alpha z}$, field $\propto e^{-\alpha z}$: power decays at \textbf{twice} the rate of $|\mathbf{E}|$.}
\end{ebox}

\vspace{0.3pt}
\shead{graybg}{gray}{ATTENUATION RATE \& DECIBELS}
\begin{ebox}
\eq{Unit conversion}{}
\[\alpha\;\text{(dB/m)} = 8.69\;\alpha\;\text{(Np/m)}\]
\eq{Power attenuation rate from $z=0$}{}
\[A = 10\log\!\left(\frac{S_\text{av}(z)}{S_\text{av}(0)}\right) = -8.69\,\alpha z\;\text{(dB)}\]
\eq{Field attenuation from $z=0$}{}
\[A = 20\log\!\left(\frac{|E(z)|}{|E(0)|}\right)\;\text{(dB)}\]
\eq{Power/voltage ratio}{}
\[G\;\text{(dB)}=10\log\!\left(\frac{P_1}{P_2}\right)=20\log\!\left(\frac{V_1}{V_2}\right)\]
\textit{$P\!\times\!2\!\Rightarrow\!+3\,\text{dB}$;\enspace$P\!\times\!10\!\Rightarrow\!+10\,\text{dB}$;\enspace$P\!\times\!100\!\Rightarrow\!+20\,\text{dB}$.}
\end{ebox}

\vspace{0.3pt}
\shead{coralbg}{coral}{CURRENT IN GOOD CONDUCTOR}
\begin{ebox}
\eq{Current density at depth $z$}{$J_0=\sigma E_0$}
\[\tilde{\mathbf{J}}(z)=J_0\,e^{-(1+j)z/\delta_s}\;\text{(A/m}^2\text{)},\qquad\delta_s=\frac{1}{\sqrt{\pi f\mu\sigma}}\;\text{(m)}\]
\eq{Total current}{slab width $w$ ($y$-dir)}
\[\tilde{I}=\frac{J_0\,w\,\delta_s}{1+j}\;\text{(A)}\]
\eq{Surface/internal impedance}{}
\[Z_s=\frac{1+j}{\sigma\delta_s}=R_s(1+j),\qquad R_s=\sqrt{\frac{\pi f\mu}{\sigma}}\;(\Omega)\]
\eq{AC resistance of slab (width $w$, length $\ell$)}{}
\[R=R_s\frac{\ell}{w}\;(\Omega)\]
\eq{Coaxial cable resistance per unit length}{}
\[R'=\frac{R_s}{2\pi}\!\left(\frac{1}{a}+\frac{1}{b}\right)\;(\Omega/\text{m})\]
\textit{$a$: inner conductor radius;\enspace$b$: outer conductor radius (m).}
\end{ebox}
