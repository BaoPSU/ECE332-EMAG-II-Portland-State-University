%-- Column 1 -----------------------------------
\shead{purplebg}{purple}{NORMAL INCIDENCE --- $\Gamma$ AND $\tau$}
\begin{ebox}
\textit{\color{coral}\textbf{Med.\ 1} ($z<0$): incident side.\enspace\textbf{Med.\ 2} ($z\ge0$): transmitted side. Boundary at $z=0$.}

\eq{Reflection coeff.}{wave from medium 1}
\[\Gamma = \frac{\eta_2-\eta_1}{\eta_2+\eta_1}\;(\text{unitless})\]

\eq{Transmission coeff.}{always}
\[\tau = 1+\Gamma = \frac{2\eta_2}{\eta_2+\eta_1}\]

\eq{Intrinsic impedance}{general $\to$ nonmag.\ shortcut}
\[\eta_i = \sqrt{\tfrac{\mu_r}{\varepsilon_r}}\,\eta_0 \xrightarrow{\mu_r=1} \tfrac{\eta_0}{\sqrt{\varepsilon_{r,i}}}\;(\Omega)\qquad \eta_0=120\pi\!\approx\!377\,\Omega\]
\textit{\color{coral}Default $\mu_r=1$ (air, dielectric, ``nonmag.''). Only use $\mu_r\!\ne\!1$ if problem says so (``ferromagnetic'', iron, ferrite, or $\mu_r$ given explicitly).}

\eq{Low-loss $\eta_2$ (poor conductor)}{$\sigma/(\omega\varepsilon)\ll1$}
\[\eta_2 \approx \sqrt{\tfrac{\mu}{\varepsilon'}}\!\left(1+j\tfrac{\sigma}{2\omega\varepsilon'}\right) \;\xrightarrow{\text{drop tiny }j}\; \tfrac{\eta_0}{\sqrt{\varepsilon_{r,2}}}\;(\Omega)\]
\textit{\color{coral}For $\Gamma$/$\tau$ just use $\eta_0/\sqrt{\varepsilon_{r,2}}$. The $j$ term is for accuracy with $|\eta_c|, \theta_\eta$.}
\end{ebox}

\vspace{2pt}
\shead{purplebg}{purple}{$\tilde{\mathbf{E}}$ PHASOR TEMPLATES}
\begin{ebox}
\eq{Wavenumber per medium}{nonmag.}
\[k_i = \tfrac{\omega\sqrt{\varepsilon_{r,i}}}{c}\;(\text{rad/m})\quad\Rightarrow\;k_2 = k_1\sqrt{\varepsilon_{r,2}/\varepsilon_{r,1}}\]

\eq{Direction table}{$u\in\{x,y,z\}$ = axis perp.\ to boundary}\\[1pt]
{\centering\makebox[0.98\linewidth][c]{{\setlength{\tabcolsep}{3pt}\begin{tabular}{@{}cccc@{}}
\toprule
Inc dir & Inc & Refl & Trans\\
\midrule
\rowcolor{rowB}$+\hat{u}$ & $e^{-jk_1u}$ & $e^{+jk_1u}$ & $e^{-jk_2u}$\\
$-\hat{u}$ & $e^{+jk_1u}$ & $e^{-jk_1u}$ & $e^{+jk_2u}$\\
\bottomrule
\end{tabular}}}\par}
\textit{$+\hat{u}$ direction $\Leftrightarrow$ Med 2 at $u\!\ge\!0$. Reflected sign flips, transmitted matches incident.}\\[2pt]

\textit{General template (incident has pol.\ $\hat{e}$, amplitude $a_0$):}
\[\tilde{\mathbf{E}}^i = a_0\hat{e}\,e^{\mp jk_1u},\quad
  \tilde{\mathbf{E}}^r = \Gamma a_0\hat{e}\,e^{\pm jk_1u},\quad
  \tilde{\mathbf{E}}^t = \tau a_0\hat{e}\,e^{\mp jk_2u}\]
\textit{(lossy med 2: replace $e^{\mp jk_2u}$ with $e^{\mp\alpha_2 u}e^{\mp j\beta_2 u}$)}\\[1pt]
\textit{Total in medium 1: $\tilde{\mathbf{E}}_1 = \tilde{\mathbf{E}}^i + \tilde{\mathbf{E}}^r$.}\\[1pt]
\eq{$\hat{e}$ by polarization}{plug into templates above}
{\centering\makebox[0.98\linewidth][c]{{\setlength{\tabcolsep}{2pt}\begin{tabular}{@{}p{0.30\linewidth}p{0.34\linewidth}p{0.30\linewidth}@{}}
\toprule
Pol. & $\hat{e}$ & Note\\
\midrule
\rowcolor{rowB}Lin.\ $\hat{x}$ & $\hat{x}$ & $E_y=0$\\
Lin.\ $\hat{y}$ & $\hat{y}$ & $E_x=0$\\
\rowcolor{rowB}Lin.\ at $45^\circ$ & $(\hat{x}+\hat{y})/\sqrt{2}$ & in phase\\
RHC & $\hat{x}-j\hat{y}$ & $\delta=-\pi/2$\\
\rowcolor{rowB}LHC & $\hat{x}+j\hat{y}$ & $\delta=+\pi/2$\\
General & $a_x\hat{x}+a_y e^{j\delta}\hat{y}$ & elliptical\\
\bottomrule
\end{tabular}}}\par}
\textit{\color{coral}$\Gamma, \tau$ act on $\hat{e}$ unchanged. Only swap the polarization, not the formulas.}
\end{ebox}

\vspace{2pt}
\shead{coralbg}{coral}{POWER REFLECTION \& TRANSMISSION}
\begin{ebox}
\eq{\% reflected}{always}
\[\%P_r = 100\,|\Gamma|^2\]
\eq{\% transmitted}{$\eta$-ratio matters!}
\[\%P_t = 100\,|\tau|^2\,\frac{\eta_1}{\eta_2}\]
\textit{Check: $\%P_r + \%P_t = 100$.}\\
\textit{\color{coral}If medium 2 is denser ($\eta_2<\eta_1$), $|\tau|>1$ but $\%P_t<100\%$ because of the $\eta_1/\eta_2$ factor.}
\end{ebox}

\vspace{2pt}
\shead{coralbg}{coral}{LOW-LOSS QUICK PARAMS (med 2)}
\begin{ebox}
\eq{Check first}{$\sigma/(\omega\varepsilon)$}\\[1pt]
\[\alpha_2 \approx \tfrac{\sigma_2}{2}\sqrt{\tfrac{\mu_2}{\varepsilon_2}} = \tfrac{\sigma_2\eta_0}{2\sqrt{\varepsilon_{r2}}}\;\text{(Np/m)}\]
\[\beta_2 \approx \omega\sqrt{\mu_2\varepsilon_2} = \tfrac{\omega\sqrt{\varepsilon_{r2}}}{c}\;\text{(rad/m)}\]
\[\eta_2 \approx \tfrac{\eta_0}{\sqrt{\varepsilon_{r2}}}\;\angle\!\approx\!0\]
\textit{$\varepsilon = \varepsilon_r\varepsilon_0$;\enspace $\varepsilon_0 = 8.85\!\times\!10^{-12}$ F/m;\enspace $36\pi\varepsilon_0\approx1\!\times\!10^{-9}$.}
\end{ebox}

\columnbreak

%-- Column 2 -----------------------------------
\shead{tealbg}{teal}{SNELL'S LAW --- OBLIQUE INCIDENCE}
\begin{ebox}
\eq{Single boundary}{angles from normal}
\[\sin\theta_2 = \sin\theta_1\sqrt{\tfrac{\varepsilon_{r1}}{\varepsilon_{r2}}} = \sin\theta_1\,\tfrac{n_1}{n_2}\]

\eq{Multilayer chain}{stack of slabs $1\to2\to3\to4$}
\[\sin\theta_{i+1} = \sin\theta_i\sqrt{\tfrac{\varepsilon_{r,i}}{\varepsilon_{r,i+1}}}\]

\eq{Air--slabs--air shortcut}{outer media identical}
\[\theta_{\text{exit}} = \theta_{\text{incident}}\]
\textit{\color{coral}Snell only sets angles. Reflection/transmission \emph{magnitudes} need Fresnel coefficients (parallel vs perpendicular).}
\end{ebox}

\vspace{2pt}
\shead{tealbg}{teal}{LATERAL DISPLACEMENT}
\begin{ebox}
\eq{Beam offset through $N$ slabs}{thickness $t_i$, refracted angle $\theta_i$}
\[d = \sum_{i=1}^{N} t_i\,\tan\theta_i\]
\textit{Each slab adds $t_i\tan\theta_i$. Final air-to-air exit returns to $\theta_1$ but is laterally shifted by $d$.}\\[1pt]
\eq{Common values}{$\arctan$}\\
{\centering\makebox[0.98\linewidth][c]{\begin{tabular}{@{}cl@{}}
\toprule
$\sin\theta$ & $\theta$\\
\midrule
\rowcolor{rowB}$0.2$ & $11.54^\circ$\\
$1/3$ & $19.47^\circ$\\
\rowcolor{rowB}$1/2$ & $30^\circ$\\
$0.5774$ & $35.26^\circ$\\
\rowcolor{rowB}$1/\sqrt{2}$ & $45^\circ$\\
\bottomrule
\end{tabular}}\par}
\end{ebox}

\vspace{2pt}
\shead{bluebg}{blue}{FRESNEL OBLIQUE --- $\Gamma, \tau, R, T$}
\begin{ebox}
\eq{Perpendicular ($E\perp$ plane of inc.)}{}
\[\Gamma_\perp = \tfrac{\eta_2\cos\theta_i-\eta_1\cos\theta_t}{\eta_2\cos\theta_i+\eta_1\cos\theta_t},\quad \tau_\perp = 1+\Gamma_\perp\]

\eq{Parallel ($E\parallel$ plane of inc.)}{}
\[\Gamma_\parallel = \tfrac{\eta_2\cos\theta_t-\eta_1\cos\theta_i}{\eta_2\cos\theta_t+\eta_1\cos\theta_i},\quad \tau_\parallel = (1+\Gamma_\parallel)\tfrac{\cos\theta_i}{\cos\theta_t}\]

\eq{Critical angle (TIR)}{only if $n_1>n_2$}
\[\sin\theta_c = \tfrac{n_2}{n_1} = \sqrt{\tfrac{\varepsilon_{r2}}{\varepsilon_{r1}}}\;\;\Rightarrow\;\theta_i\!\ge\!\theta_c\!=\!\text{total internal refl.}\]

\eq{Brewster angle ($\parallel$ only)}{$\Gamma_\parallel=0$, no reflection}
\[\tan\theta_{B\parallel} = \sqrt{\tfrac{\varepsilon_{r2}}{\varepsilon_{r1}}}\;\text{(nonmag.)}\]

\eq{Reflectivity / Transmissivity}{power form; $R+T=1$}
\[R = |\Gamma|^2,\quad T = |\tau|^2\tfrac{\eta_1\cos\theta_t}{\eta_2\cos\theta_i}\]
\textit{Normal incidence: $\theta_t=\theta_i=0$, cos factors drop, $R = |\Gamma|^2$, $T = |\tau|^2(\eta_1/\eta_2)$.}
\end{ebox}

\vspace{2pt}
\shead{amberbg}{amber}{FIBER OPTICS}
\begin{ebox}
\textit{Core index $n_f$, cladding $n_c$, surrounded by $n_0$. Need $n_f>n_c$ for TIR.}

\eq{Index of refraction}{nonmag.\ $\mu_r=1$}
\[n = c/u_p = \sqrt{\varepsilon_r}\]

\eq{Acceptance angle $\theta_a$}{max $\theta_i$ for TIR in fiber}
\[\sin\theta_a = \tfrac{1}{n_0}\sqrt{n_f^2 - n_c^2}\]

\eq{Modal dispersion (max time delay)}{fiber length $\ell$}
\[\tau = \tfrac{\ell n_f}{c}\!\cdot\!\tfrac{n_f-n_c}{n_c}\]

\eq{Max data rate}{$T\!\ge\!2\tau$ to avoid pulse overlap}
\[f_p = \tfrac{1}{2\tau} = \tfrac{c\,n_c}{2\ell n_f(n_f-n_c)}\]
\end{ebox}

\columnbreak

%-- Column 3 -----------------------------------
\shead{coralbg}{coral}{RECTANGULAR WAVEGUIDE --- MODES}
\begin{ebox}
\eq{TE mode}{$E_z=0$, $H_z\ne0$}\\
\eq{TM mode}{$H_z=0$, $E_z\ne0$}\\[1pt]
\textit{Dimensions $a\times b$ with $a>b$ along $\hat{x},\hat{y}$; wave in $\hat{z}$.}\\[1pt]
\eq{$E_x$ form (TE)}{read $m,n$ from arguments}
\[E_x \propto \cos\!\left(\tfrac{m\pi x}{a}\right)\sin\!\left(\tfrac{n\pi y}{b}\right)\sin(\omega t-\beta z)\]
\textit{coef. on $x$ = $m\pi/a$; coef. on $y$ = $n\pi/b$.}\\[1pt]
\textit{\color{coral}TE mode allowed if at least one of $m,n\ge1$. TM needs both $m,n\ge1$.}
\end{ebox}

\vspace{2pt}
\shead{coralbg}{coral}{CUTOFF FREQUENCY}
\begin{ebox}
\eq{$f_{mn}$}{$m,n$ integers $\ge0$ (not both zero)}
\[f_{mn} = \tfrac{u_{p0}}{2}\sqrt{\!\left(\tfrac{m}{a}\right)^{\!2}\!+\!\left(\tfrac{n}{b}\right)^{\!2}}\;\text{(Hz)}\]
\[u_{p0} = \tfrac{c}{\sqrt{\varepsilon_r}}\;\text{(m/s, guide medium)}\]

\eq{Common cutoffs ($a>b$)}{}\\
{\centering\makebox[0.98\linewidth][c]{{\setlength{\tabcolsep}{2pt}\begin{tabular}{@{}p{0.20\linewidth}p{0.70\linewidth}@{}}
\toprule
Mode & $f_c$\\
\midrule
\rowcolor{rowB}TE$_{10}$ & $u_{p0}/(2a)$ (dominant)\\
TE$_{01}$ & $u_{p0}/(2b)$\\
\rowcolor{rowB}TE$_{20}$ & $u_{p0}/a = 2f_{10}$\\
TE$_{11}$,TM$_{11}$ & $\tfrac{u_{p0}}{2}\sqrt{1/a^2+1/b^2}$\\
\bottomrule
\end{tabular}}}\par}
\textit{Propagation requires $f>f_{mn}$.}
\end{ebox}

\vspace{2pt}
\shead{coralbg}{coral}{$\varepsilon_r$ FROM DISPERSION}
\begin{ebox}
\eq{Given $\omega$, $\beta$, mode $(m,n)$}{}
\[\varepsilon_r = \tfrac{c^2}{\omega^2}\!\left[\beta^2 + \!\left(\tfrac{m\pi}{a}\right)^{\!2}\!+\!\left(\tfrac{n\pi}{b}\right)^{\!2}\right]\]
\textit{Sanity: $\beta$ in rad/m, $a,b$ in m, $\omega$ in rad/s.}
\end{ebox}

\vspace{2pt}
\shead{coralbg}{coral}{WAVE IMPEDANCE \& $H_y$}
\begin{ebox}
\[Z_{\text{TE}} = \tfrac{\eta}{\sqrt{1-(f_c/f)^2}}\,(\Omega)\qquad Z_{\text{TM}} = \eta\sqrt{1-(f_c/f)^2}\]
\[\eta = \tfrac{\eta_0}{\sqrt{\varepsilon_r}} \qquad H_y = \tfrac{E_x}{Z_{\text{TE}}}\;\text{(TE mode)}\]
\textit{$Z_{\text{TE}}>\eta$ always (denominator $<1$); $Z_{\text{TM}}<\eta$.}
\end{ebox}

\vspace{2pt}
\shead{amberbg}{amber}{GROUP VELOCITY \& TRAVEL TIME}
\begin{ebox}
\[u_g = u_{p0}\sqrt{1-(f_{mn}/f)^2}\;\text{(m/s)}\]
\[u_p = \tfrac{u_{p0}}{\sqrt{1-(f_{mn}/f)^2}}\quad\Rightarrow\quad u_pu_g = u_{p0}^2\]
\[t = L/u_g\;\text{(s)}\quad\text{travel time through guide of length }L\]
\textit{Higher modes have lower $u_g$ $\Rightarrow$ they arrive later. Group delay spread = sum over excited modes.}
\end{ebox}

\shead{purplebg}{purple}{DIPOLE TYPE BY LENGTH}
\begin{ebox}
\eq{Check first}{compute $\lambda=c/f$, then $l/\lambda$}\\[1pt]
{\centering\makebox[0.98\linewidth][c]{{\setlength{\tabcolsep}{2pt}\begin{tabular}{@{}p{0.30\linewidth}p{0.30\linewidth}p{0.32\linewidth}@{}}
\toprule
$l/\lambda$ & Type & Use\\
\midrule
\rowcolor{rowB}$<1/50$ & Hertzian (short) & Hertzian formula\\
$=1/4$ & Quarter-wave & Arbitrary formula\\
\rowcolor{rowB}$=1/2$ & Half-wave & $D=1.64$, simple\\
other & Arbitrary & Arbitrary formula\\
\bottomrule
\end{tabular}}}\par}
\textit{\color{coral}If $l/\lambda<1/50$ stop and use Hertzian only --- don't subst.\ small $l$ into arbitrary formula.}
\end{ebox}

\vspace{2pt}
\shead{purplebg}{purple}{HERTZIAN DIPOLE --- $S(R,\theta)$}
\begin{ebox}
\eq{Power density}{$l/\lambda<1/50$; far field}
\[S(R,\theta) = \tfrac{\eta_0 k^2 I_0^2 l^2}{32\pi^2 R^2}\sin^2\theta\;\text{(W/m}^2\text{)}\]
\textit{$k=2\pi/\lambda$ rad/m;\enspace $I_0$ peak current (A);\enspace $l$ length (m);\enspace $R$ obs.\ dist (m).}\\
\eq{Max direction}{$\theta=\pi/2$ (broadside)}\\
\eq{Total radiated power}{integrate over $4\pi$ sr}
\[P_{\text{rad}} = \tfrac{\eta_0 \pi}{3}\!\left(\tfrac{I_0 l}{\lambda}\right)^{\!2}\;\text{(W)}\]
\eq{Directivity}{Hertzian, exact}
\[D_{\text{Hertz}} = 1.5 \quad(\text{i.e. }1.76\text{ dB})\]
\end{ebox}

\vspace{2pt}
\shead{tealbg}{teal}{ARBITRARY-LENGTH DIPOLE --- $S(\theta)$}
\begin{ebox}
\eq{Power density at distance $R$}{any $l$}
\[S(\theta) = \tfrac{15 I_0^2}{\pi R^2}\!\left[\tfrac{\cos\!\left(\tfrac{\pi l}{\lambda}\cos\theta\right)-\cos\!\left(\tfrac{\pi l}{\lambda}\right)}{\sin\theta}\right]^{\!2}\]

\eq{Normalized pattern}{$F(\theta)=S(\theta)/S_{\max}$, dimensionless}\\[1pt]

\eq{Half-wave $l=\lambda/2$}{$\pi l/\lambda=\pi/2$}
\[S(\theta) = \tfrac{15 I_0^2}{\pi R^2}\tfrac{\cos^2(\tfrac{\pi}{2}\cos\theta)}{\sin^2\theta},\quad D=1.64,\;R_{\text{rad}}=73\,\Omega\]

\eq{Quarter-wave $l=\lambda/4$}{$\pi l/\lambda=\pi/4$, $\cos(\pi/4)=\sqrt{2}/2$}
\[\tfrac{S(\theta)}{S_0}\!=\!\left[\tfrac{\cos(\tfrac{\pi}{4}\cos\theta)-\tfrac{\sqrt{2}}{2}}{\sin\theta}\right]^{\!2},\;S_0=\tfrac{15I_0^2}{\pi R^2}\]
\[S_{\max} = S(\pi/2) = S_0\!\left(1-\tfrac{\sqrt{2}}{2}\right)^{\!2}\!\approx 0.0858\,S_0\]
\end{ebox}

\columnbreak

%-- Column 2 -----------------------------------
\shead{greenbg}{green}{BEAM PARAMETERS --- HPBW, $\Omega_p$, $D$}
\begin{ebox}
\eq{1. Normalize}{}
\[F(\theta) = S(\theta)/S_{\max}\]

\eq{2. Half-power beamwidth (HPBW)}{solve $F(\theta_{1/2})=0.5$}
\[\beta = 2\theta_{1/2}\quad\text{(symmetric beams)}\]
\eq{example}{$F(\theta)=e^{-20\theta^2}$ $\Rightarrow$ $\theta_{1/2}=\sqrt{-\ln 0.5/20}=0.186$ rad $\Rightarrow$ $\beta=21.31^\circ$}\\[1pt]

\eq{3. Pattern solid angle}{(sr)}
\[\Omega_p = \!\int_0^{2\pi}\!\!\int_0^{\pi}\!F(\theta)\sin\theta\,d\theta\,d\phi\]
\textit{If $F$ has no $\phi$ dep.: $\Omega_p = 2\pi\!\int_0^\pi F(\theta)\sin\theta\,d\theta$.}

\eq{4. Directivity}{}
\[D = \tfrac{4\pi}{\Omega_p}\;\text{(unitless)} \qquad D_{\text{dB}} = 10\log_{10}D\]
\eq{5. Gain}{$\xi$ = radiation efficiency, $0<\xi\le1$}
\[G = \xi D \qquad G_{\text{dB}} = 10\log_{10}G\]
\textit{\color{coral}For lossless / ideal antennas: $G=D$.}
\end{ebox}

\begin{ebox}

\vspace{2pt}
\shead{amberbg}{amber}{EFFECTIVE AREA $A_e$}
\begin{ebox}
\eq{Definition}{$\text{m}^2$, antenna's receiving cross section}
\[A_e = \tfrac{\lambda^2 D}{4\pi}\]
\eq{Physical cross section}{half-wave dipole, wire diameter $d$}
\[A_p = l\,d = \tfrac{\lambda}{2}\,d\]
\eq{Ratio}{$A_e\gg A_p$ for thin wires (good!)}\\
\textit{Example: $f\!=\!100$ MHz, $d\!=\!2$ cm $\Rightarrow$ $A_e/A_p\!\approx\!39$.}
\end{ebox}

\vspace{2pt}
\shead{greenbg}{green}{POWER FLOW IDENTITIES}
\begin{ebox}
\[P_{\text{rad}} = S_{\max}\,\Omega_p\,R^2 = \tfrac{4\pi R^2 S_{\max}}{D}\]
\[S_{\max} = \tfrac{P_{\text{rad}}\,D}{4\pi R^2}\;\text{(W/m}^2\text{)}\]
\textit{For isotropic radiator: $S = P_{\text{rad}}/(4\pi R^2)$.}
\end{ebox}

\columnbreak

%-- Column 3 -----------------------------------
\shead{pinkbg}{pink}{FRIIS TRANSMISSION FORMULA}
\begin{ebox}
\eq{Far-field link budget}{$P_t$ tx, $P_r$ rx, dist $R$}
\[\boxed{P_r = G_t G_r\!\left(\tfrac{\lambda}{4\pi R}\right)^{\!2}\!P_t}\]
\textit{$G_t, G_r$ \emph{linear} gains;\enspace $\lambda=c/f$;\enspace $R$ in m;\enspace $P_t, P_r$ in W.}

\eq{Solve for $R$}{}
\[R = \tfrac{\lambda}{4\pi}\sqrt{\tfrac{G_tG_rP_t}{P_r}}\]

\eq{Solve for $G_r$ given $A_e$}{}
\[G_r = \tfrac{4\pi A_{e,r}}{\lambda^2}\quad\Rightarrow\quad P_r=\tfrac{G_t P_t A_{e,r}}{4\pi R^2}\]

\textit{\color{coral}Use linear $G$, not dB. Convert first: $G_{\text{lin}}=10^{G_{\text{dB}}/10}$.}\\
\textit{Free-space path loss $L_{\text{fs}}=(4\pi R/\lambda)^2$ so $P_r = G_tG_rP_t/L_{\text{fs}}$.}
\end{ebox}

\shead{bluebg}{blue}{RADAR EQUATION (BONUS)}
\begin{ebox}
\eq{Monostatic radar}{tx = rx, target cross section $\sigma_t$ (m$^2$)}
\[P_r = \tfrac{P_t G^2 \lambda^2 \sigma_t}{(4\pi)^3 R^4}\]
