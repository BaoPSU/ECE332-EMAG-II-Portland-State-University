\vspace{1pt}
\shead{purplebg}{purple}{HERTZIAN DIPOLE --- $S(R,\theta)$\ \ (P1)}
\begin{ebox}
\eq{Power density}{$l/\lambda<1/50$; far field}
\[S(R,\theta) = \tfrac{\eta_0 k^2 I_0^2 l^2}{32\pi^2 R^2}\sin^2\theta\;(\text{W/m}^2)\]
\textit{$k\!=\!2\pi/\lambda$ (rad/m);\, $I_0$ peak current (A);\, $l$ length (m);\, $R$ obs.\ dist (m);\, $\eta_0\!=\!120\pi\!\approx\!377\,\Omega$.}\\[1pt]
\eq{Max direction}{broadside}\ \ $\theta_{\max}\!=\!\pi/2$\\[1pt]
\eq{Recipe}{}
1.\ Compute $\lambda\!=\!c/f$. Check $l/\lambda\!<\!1/50$.\\
2.\ Compute $k\!=\!2\pi/\lambda$.\\
3.\ Plug into $S(R,\theta)$ with given $R$, $\theta$.\\[1pt]
\textit{\color{blue}HW4 P1: $l\!=\!1$m, $f\!=\!1$MHz, $I_0\!=\!12$A, $R\!=\!5$km, $\theta\!=\!45^\circ$. $\lambda\!=\!300$m, $l/\lambda\!=\!1/300\!<\!1/50$ \checkmark. $k\!=\!2\pi/300$. $S\!=\!\frac{(120\pi)(2\pi/300)^2(144)(1)}{32\pi^2(5{\times}10^3)^2}\!\cdot\!\frac{1}{2}\!=\!\mathbf{1.51{\times}10^{-9}}$ W/m$^2$.}
\end{ebox}

\vspace{1pt}
\shead{tealbg}{teal}{ARBITRARY-LENGTH DIPOLE --- $S(\theta)$\ \ (P3)}
\begin{ebox}
\eq{Power density at $R$}{any $l$}
\[S(\theta) = \tfrac{15 I_0^2}{\pi R^2}\!\left[\tfrac{\cos\!\left(\tfrac{\pi l}{\lambda}\cos\theta\right)-\cos\!\left(\tfrac{\pi l}{\lambda}\right)}{\sin\theta}\right]^{\!2}\]

\eq{Quarter-wave $l\!=\!\lambda/4$}{$\pi l/\lambda\!=\!\pi/4$, $\cos(\pi/4)\!=\!\sqrt{2}/2$}
\[\tfrac{S(\theta)}{S_0}\!=\!\left[\tfrac{\cos(\tfrac{\pi}{4}\cos\theta)-\tfrac{\sqrt{2}}{2}}{\sin\theta}\right]^{\!2},\;S_0\!=\!\tfrac{15I_0^2}{\pi R^2}\]
\[S_{\max} = S(\pi/2) = S_0\!\left(1-\tfrac{\sqrt{2}}{2}\right)^{\!2}\!\approx 0.0858\,S_0\]
\eq{Normalized pattern}{plot this for P3c}
\[F(\theta) = \tfrac{S(\theta)}{S_{\max}} = \tfrac{1}{0.0858}\!\left[\tfrac{\cos(\tfrac{\pi}{4}\cos\theta)-\tfrac{\sqrt{2}}{2}}{\sin\theta}\right]^{\!2}\]
\textit{\color{blue}HW4 P3: $\theta_{\max}\!=\!\pi/2$ (broadside);\ \ $S_{\max}\!=\!0.0858\,S_0$;\ \ $F(\theta)$ peaks $\!=\!1$ at $\pi/2$, drops to $0$ at $\theta\!=\!0,\pi$.}\\[1pt]

\eq{Half-wave $l\!=\!\lambda/2$ (ref.\ only)}{$\pi l/\lambda\!=\!\pi/2$}
\[S(\theta) = \tfrac{15 I_0^2}{\pi R^2}\tfrac{\cos^2(\tfrac{\pi}{2}\cos\theta)}{\sin^2\theta},\;D\!=\!1.64,\,R_{\text{rad}}\!=\!73\,\Omega\]
\end{ebox}

%-- Column 2 -----------------------------------
\shead{greenbg}{green}{BEAM PARAMETERS --- $\beta$, $\Omega_p$, $D$\ \ (P2)}
\begin{ebox}
\eq{Step 1 --- Normalize}{always start here}
\[F(\theta) = S(\theta)/S_{\max}\]

\eq{Step 2 --- HPBW $\beta$}{solve $F(\theta_{1/2})\!=\!0.5$, then $\beta\!=\!2\theta_{1/2}$ for symmetric beams}\\[1pt]
\textit{\color{coral}Gaussian shortcut: $F\!=\!e^{-a\theta^2}\!=\!0.5\Rightarrow\theta_{1/2}\!=\!\sqrt{\ln 2/a}$.}\\
\textit{\color{blue}HW4 P2a: $F\!=\!e^{-20\theta^2}$, $a\!=\!20\Rightarrow\theta_{1/2}\!=\!\sqrt{\ln 2/20}\!=\!0.186$ rad. $\boxed{\beta\!=\!0.372\text{ rad}\!=\!21.31^\circ}$.}\\[1pt]

\eq{Step 3 --- Pattern solid angle $\Omega_p$}{units: sr}
\[\Omega_p = \!\int_0^{2\pi}\!\!\int_0^{\pi}\!F(\theta)\sin\theta\,d\theta\,d\phi\]
\textit{No $\phi$ dep.: $\Omega_p\!=\!2\pi\!\int_0^\pi\!F(\theta)\sin\theta\,d\theta$.}\\
\textit{\color{coral}Most integrals have NO closed form. Use Wolfram / numerical.}\\
\textit{Wolfram: \texttt{2*pi*Integrate[Exp[-20 t\textasciicircum 2]*Sin[t],\{t,0,pi\}]}}\\
\textit{\color{blue}HW4 P2b: $\boxed{\Omega_p\!=\!0.156\text{ sr}}$ (numeric).}\\[1pt]

\eq{Step 4 --- Directivity $D$}{}
\[D = \tfrac{4\pi}{\Omega_p}\quad(\text{unitless})\qquad D_{\text{dB}}\!=\!10\log_{10}D\]
\textit{\color{blue}HW4 P2c: $D\!=\!4\pi/0.156\!=\!\boxed{80.55}$ ($19.06$ dB).}\\[1pt]

\eq{Step 5 --- Gain (if asked)}{$\xi\!=\!$ rad.\ eff., $0<\xi\le1$}
\[G = \xi D\qquad G_{\text{dB}}\!=\!10\log_{10}G\]
\textit{Lossless / ideal antenna $\Rightarrow$ $G\!=\!D$.}
\end{ebox}

\vspace{1pt}
\shead{amberbg}{amber}{EFFECTIVE AREA $A_e$\ \ (P4)}
\begin{ebox}
\eq{Definition}{antenna's RX cross section (m$^2$)}
\[A_e = \tfrac{\lambda^2 D}{4\pi}\;(\text{m}^2)\]
\eq{Physical cross section}{wire dipole, diameter $d$}
\[A_p = l\,d = \tfrac{\lambda}{2}\,d\;\text{(half-wave)}\]
\eq{Recipe}{}
1.\ $\lambda\!=\!c/f$.\quad 2.\ Pick $D$ (1.64 for half-wave).\\
3.\ $A_e\!=\!\lambda^2 D/(4\pi)$.\quad 4.\ $A_p\!=\!l d$.\quad 5.\ Ratio.\\[1pt]
\textit{\color{blue}HW4 P4: half-wave, $f\!=\!100$MHz, $d\!=\!2$cm. $\lambda\!=\!3$m, $D\!=\!1.64$. $A_e\!=\!\frac{9(1.64)}{4\pi}\!=\!\boxed{1.17\text{ m}^2}$. $A_p\!=\!(1.5)(0.02)\!=\!0.03$ m$^2$. Ratio $\boxed{A_e/A_p\!=\!39.2}$.}\\
\textit{\color{coral}Thin wire: $A_e\!\gg\!A_p$. Antenna captures field over $\sim\!\lambda$.}
\end{ebox}

%-- Column 3 -----------------------------------
\shead{pinkbg}{pink}{FRIIS TRANSMISSION FORMULA\ \ (P5)}
\begin{ebox}
\eq{Far-field link budget}{$P_t$ tx, $P_r$ rx, range $R$}
\[\boxed{P_r = G_t G_r\!\left(\tfrac{\lambda}{4\pi R}\right)^{\!2}\!P_t}\;(\text{W})\]
\textit{$G_t, G_r$ are \emph{linear} gains;\, $\lambda\!=\!c/f$;\, $R$ in m;\, $P_t, P_r$ in W.}\\[1pt]
\eq{Solve for $R$}{max range}
\[R = \tfrac{\lambda}{4\pi}\sqrt{\tfrac{G_tG_rP_t}{P_r}}\;(\text{m})\]
\eq{Equivalent form using $A_e$}{recv.\ area form}
\[P_r = \tfrac{G_t P_t A_{e,r}}{4\pi R^2},\qquad G_r = \tfrac{4\pi A_{e,r}}{\lambda^2}\]
\textit{Free-space path loss: $L_{\text{fs}}\!=\!(4\pi R/\lambda)^2$, so $P_r\!=\!G_tG_rP_t/L_{\text{fs}}$.}\\[1pt]
\eq{Recipe}{}
1.\ $\lambda\!=\!c/f$.\\
2.\ Convert any dB gains to linear: $G\!=\!10^{G_{\text{dB}}/10}$.\\
3.\ If TX is ``half-wave dipole'' use $G_t\!=\!1.64$.\\
4.\ Plug into Friis.\\[1pt]
\textit{\color{coral}Use \emph{linear} $G$, not dB. Convert FIRST.}\\
\textit{\color{blue}HW4 P5: half-wave TX, $P_t\!=\!1$kW, $f\!=\!50$MHz, RX $G_r\!=\!3$dB$\!=\!2$, $R\!=\!30$km. $\lambda\!=\!6$m, $G_t\!=\!1.64$. $P_r\!=\!(1.64)(2)\!\left(\tfrac{6}{4\pi\!\cdot\!3{\times}10^4}\right)^{\!2}\!10^3\!=\!\boxed{8.3{\times}10^{-7}\text{ W}}\!=\!0.83\,\mu$W.}
\end{ebox}

\vspace{1pt}
\shead{pinkbg}{pink}{dB $\leftrightarrow$ LINEAR}
\begin{ebox}
\[G_{\text{dB}} = 10\log_{10}G_{\text{lin}}\qquad G_{\text{lin}} = 10^{G_{\text{dB}}/10}\]
{\centering\makebox[0.98\linewidth][c]{{\setlength{\tabcolsep}{2pt}\begin{tabular}{@{}cc@{\quad}cc@{}}
\toprule
dB & linear & dB & linear\\
\midrule
\rowcolor{rowB}0 & 1 & 10 & 10\\
3 & 2 & 13 & 20\\
\rowcolor{rowB}6 & 4 & 20 & 100\\
7 & 5 & 30 & 1000\\
\rowcolor{rowB}$-3$ & $0.5$ & $-10$ & $0.1$\\
\bottomrule
\end{tabular}}}\par}
\textit{Memorize: 3 dB $\!\Leftrightarrow\!\times2$, 10 dB $\!\Leftrightarrow\!\times10$. Add dB $=$ multiply linear.}\\
\textit{\color{coral}For E-field / amplitude use $20\log$ not $10\log$ (power $\propto$ E$^2$).}
\end{ebox}

\vspace{1pt}
\shead{graybg}{gray}{SYMBOL REFERENCE}
\begin{ebox}
$l$ --- antenna length (m); $\lambda\!=\!c/f$ (m); $c\!=\!3\!\times\!10^8$ m/s\\
$I_0$ --- peak current (A); $\eta_0\!=\!120\pi\!\approx\!377\,\Omega$\\
$k\!=\!2\pi/\lambda$ wavenumber (rad/m)\\
$R$ --- range from antenna to obs.\ pt (m)\\
$S(R,\theta)$ --- power density (W/m$^2$)\\
$F(\theta)\!=\!S/S_{\max}$ --- normalized pattern (unitless)\\
$\beta$ --- half-power beamwidth (rad or deg)\\
$\Omega_p$ --- pattern solid angle (sr)\\
$D\!=\!4\pi/\Omega_p$ --- directivity (unitless)\\
$\xi$ --- radiation efficiency ($0\!<\!\xi\!\le\!1$); $G\!=\!\xi D$\\
$A_e\!=\!\lambda^2 D/(4\pi)$ --- effective area (m$^2$)\\
$A_p\!=\!l d$ --- physical cross sec.\ (m$^2$)\\
$P_t, P_r$ --- TX / RX power (W); $G_t,G_r$ linear gains
\end{ebox}
