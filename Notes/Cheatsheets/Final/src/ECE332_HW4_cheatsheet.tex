\begin{ebox}

\midrule

\vspace{1pt}
\shead{purplebg}{purple}{HERTZIAN DIPOLE --- $S(R,\theta)$}
\begin{ebox}
\eq{Far-field E,H phasors}{small $l\!\ll\!\lambda$}
\[\tilde{E}_\theta = \tfrac{j I_0 k l \eta_0}{4\pi}\tfrac{e^{-jkR}}{R}\sin\theta,\;\; \tilde{H}_\phi = \tilde{E}_\theta/\eta_0\]
\eq{Power density}{$l/\lambda<1/50$; far field}
\[S(R,\theta) = \tfrac{\eta_0 k^2 I_0^2 l^2}{32\pi^2 R^2}\sin^2\theta = S_{\max}\sin^2\theta\,(\text{W/m}^2)\]
\textit{$k\!=\!2\pi/\lambda$ (rad/m);\, $I_0$ peak current (A);\, $l$ length (m);\, $R$ obs.\ dist (m);\, $\eta_0\!=\!120\pi\!\approx\!377\,\Omega$.}\\[1pt]
\eq{Max direction}{broadside}\ \ $\theta_{\max}\!=\!\pi/2$\\
\eq{Total radiated power}{integrate over $4\pi$}
\[P_{\text{rad}} = \tfrac{\eta_0 \pi}{3}\!\left(\tfrac{I_0 l}{\lambda}\right)^{\!2}\,(\text{W})\]
\eq{Directivity}{exact}\ \ $D_{\text{Hertz}}\!=\!1.5$ ($1.76$ dB)
\end{ebox}

\vspace{1pt}
\shead{tealbg}{teal}{ARBITRARY-LENGTH DIPOLE --- $S(\theta)$}
\begin{ebox}
\eq{Power density at $R$}{any $l$}
\[S(\theta) = \tfrac{15 I_0^2}{\pi R^2}\!\left[\tfrac{\cos\!\left(\tfrac{\pi l}{\lambda}\cos\theta\right)-\cos\!\left(\tfrac{\pi l}{\lambda}\right)}{\sin\theta}\right]^{\!2}\]
\eq{Normalized pattern}{dimensionless}\ \ $F(\theta)\!=\!S(\theta)/S_{\max}$\\[1pt]

\eq{Quarter-wave $l\!=\!\lambda/4$}{$\pi l/\lambda\!=\!\pi/4$, $\cos(\pi/4)\!=\!\sqrt{2}/2$}
\[\tfrac{S(\theta)}{S_0}\!=\!\left[\tfrac{\cos(\tfrac{\pi}{4}\cos\theta)-\tfrac{\sqrt{2}}{2}}{\sin\theta}\right]^{\!2},\;S_0\!=\!\tfrac{15I_0^2}{\pi R^2}\]
\[S_{\max} = S(\pi/2) = S_0\!\left(1-\tfrac{\sqrt{2}}{2}\right)^{\!2}\!\approx 0.0858\,S_0\]

\eq{Half-wave $l\!=\!\lambda/2$}{$\pi l/\lambda\!=\!\pi/2$, shortcut form}
\[S(\theta) = \tfrac{15 I_0^2}{\pi R^2}\tfrac{\cos^2(\tfrac{\pi}{2}\cos\theta)}{\sin^2\theta}\]
$D\!=\!1.64$, $R_{\text{rad}}\!=\!73\,\Omega$, $\theta_{\max}\!=\!\pi/2$.
\end{ebox}

\columnbreak

%-- Column 2 -----------------------------------
\shead{greenbg}{green}{BEAM PARAMETERS --- $\beta$, $\Omega_p$, $D$}
\begin{ebox}
\eq{Step 1 --- Normalize}{always start here}
\[F(\theta) = S(\theta)/S_{\max}\;(\text{unitless})\]

\eq{Step 2 --- HPBW $\beta$}{solve $F(\theta_{1/2})\!=\!0.5$, then $\beta\!=\!2\theta_{1/2}$ for symmetric beams}\\[1pt]
\textit{\color{coral}Gaussian shortcut: $F\!=\!e^{-a\theta^2}\!=\!0.5\Rightarrow\theta_{1/2}\!=\!\sqrt{\ln 2/a}$.}\\[1pt]

\eq{Step 3 --- Pattern solid angle $\Omega_p$}{units: sr}
\[\Omega_p = \!\int_0^{2\pi}\!\!\int_0^{\pi}\!F(\theta)\sin\theta\,d\theta\,d\phi\]
\textit{No $\phi$ dep.: $\Omega_p\!=\!2\pi\!\int_0^\pi\!F(\theta)\sin\theta\,d\theta$.}\\
\textit{\color{coral}Most integrals have NO closed form. Use Wolfram / numerical.}\\
\textit{Wolfram: \texttt{2*pi*Integrate[F[t]*Sin[t],\{t,0,pi\}]}}\\[1pt]

\eq{Step 4 --- Directivity $D$}{}
\[D = \tfrac{4\pi}{\Omega_p}\quad(\text{unitless})\qquad D_{\text{dB}}\!=\!10\log_{10}D\]
\textit{Alt.\ with two orthogonal HPBWs: $D\!\approx\!4\pi/(\beta_{xz}\beta_{yz})$.}\\[1pt]

\eq{Step 5 --- Gain (if asked)}{$\xi\!=\!$ rad.\ eff., $0<\xi\le1$}
\[G = \xi D\qquad G_{\text{dB}}\!=\!10\log_{10}G\]
\textit{Lossless / ideal antenna $\Rightarrow$ $G\!=\!D$.}
\end{ebox}

\vspace{1pt}
\shead{amberbg}{amber}{EFFECTIVE AREA $A_e$}
\begin{ebox}
\eq{Definition}{antenna's RX cross section (m$^2$)}
\[A_e = \tfrac{\lambda^2 D}{4\pi}\;(\text{m}^2)\]
\eq{Physical cross section}{wire dipole, diameter $d$}
\[A_p = l\,d = \tfrac{\lambda}{2}\,d\;\text{(half-wave)}\]
\eq{Inverse: $D$ from $A_e$}{}
\[D = \tfrac{4\pi A_e}{\lambda^2}\;(\text{unitless})\]
\textit{\color{coral}Thin wire: $A_e\!\gg\!A_p$. Antenna captures field over $\sim\!\lambda$.}
\end{ebox}

\vspace{1pt}
\shead{coralbg}{coral}{$R_{\text{rad}}$, EFFICIENCY, POWER SPLIT}
\begin{ebox}
\eq{Radiation resistance}{$\Omega$, equiv.\ ckt.\ load absorbing $P_{\text{rad}}$}
\[R_{\text{rad}} = \tfrac{2 P_{\text{rad}}}{I_0^2}\quad R_{\text{loss}} = \tfrac{2 P_{\text{loss}}}{I_0^2}\]
\eq{Input power split}{$P_t$ from generator}
\[P_t = P_{\text{rad}} + P_{\text{loss}}\quad R_{\text{in}} = R_{\text{rad}} + R_{\text{loss}}\]
\eq{Radiation efficiency}{two equivalent forms}
\[\xi = \tfrac{P_{\text{rad}}}{P_t} = \tfrac{R_{\text{rad}}}{R_{\text{rad}}+R_{\text{loss}}}\;(\text{unitless})\]
\end{ebox}

%-- Column 3 -----------------------------------
\shead{pinkbg}{pink}{FRIIS TRANSMISSION FORMULA}
\begin{ebox}
\eq{Far-field link budget}{$P_t$ tx, $P_r$ rx, range $R$}
\[\boxed{P_r = G_t G_r\!\left(\tfrac{\lambda}{4\pi R}\right)^{\!2}\!P_t}\;(\text{W})\]
\textit{$G_t, G_r$ are \emph{linear} gains;\, $\lambda\!=\!c/f$;\, $R$ in m;\, $P_t, P_r$ in W.}\\[1pt]
\eq{Solve for $R$}{max range}
\[R = \tfrac{\lambda}{4\pi}\sqrt{\tfrac{G_tG_rP_t}{P_r}}\;(\text{m})\]
\eq{Equivalent form using $A_e$}{recv.\ area form}
\[P_r = \tfrac{G_t P_t A_{e,r}}{4\pi R^2},\quad G_r = \tfrac{4\pi A_{e,r}}{\lambda^2}\]
\textit{Free-space path loss: $L_{\text{fs}}\!=\!(4\pi R/\lambda)^2$, so $P_r\!=\!G_tG_rP_t/L_{\text{fs}}$.}\\[1pt]
\eq{Recipe}{}
1.\ $\lambda\!=\!c/f$.\ \ 2.\ Convert dB gains to linear: $G\!=\!10^{G_{\text{dB}}/10}$.\\
3.\ If TX is ``half-wave dipole'' use $G_t\!=\!1.64$.\ \ 4.\ Plug into Friis.\\[1pt]
\textit{\color{coral}Use \emph{linear} $G$, not dB. Convert FIRST.}
\end{ebox}

\vspace{1pt}
\shead{pinkbg}{pink}{dB $\leftrightarrow$ LINEAR}
\begin{ebox}
\[G_{\text{dB}} = 10\log_{10}G_{\text{lin}}\qquad G_{\text{lin}} = 10^{G_{\text{dB}}/10}\]
{\centering\makebox[0.98\linewidth][c]{{\setlength{\tabcolsep}{2pt}\begin{tabular}{@{}cc@{\quad}cc@{}}
\toprule
dB & linear & dB & linear\\
\midrule
\rowcolor{rowB}0 & 1 & 10 & 10\\
3 & 2 & 13 & 20\\
\rowcolor{rowB}6 & 4 & 20 & 100\\
7 & 5 & 30 & 1000\\
\rowcolor{rowB}$-3$ & $0.5$ & $-10$ & $0.1$\\
\bottomrule
\end{tabular}}}\par}
\textit{Memorize: 3 dB $\!\Leftrightarrow\!\times2$, 10 dB $\!\Leftrightarrow\!\times10$. Add dB $=$ multiply linear.}\\
\textit{\color{coral}For E-field / amplitude use $20\log$ not $10\log$ (power $\propto$ E$^2$).}
\end{ebox}

\vspace{1pt}
\shead{greenbg}{green}{POWER FLOW IDENTITIES}
\begin{ebox}
\[P_{\text{rad}} = S_{\max}\,\Omega_p\,R^2 = \tfrac{4\pi R^2 S_{\max}}{D}\,(\text{W})\]
\[S_{\max} = \tfrac{P_{\text{rad}}\,D}{4\pi R^2}\,(\text{W/m}^2)\]
\textit{Isotropic radiator: $S\!=\!P_{\text{rad}}/(4\pi R^2)$ (no $D$ factor).}
\end{ebox}

\vspace{1pt}
\shead{graybg}{gray}{SYMBOL REFERENCE}
\begin{ebox}
$l$ --- antenna length (m); $\lambda\!=\!c/f$ (m); $c\!=\!3\!\times\!10^8$ m/s\\
$I_0$ --- peak current (A); $\eta_0\!=\!120\pi\!\approx\!377\,\Omega$\\
$k\!=\!2\pi/\lambda$ wavenumber (rad/m); $R$ obs.\ dist (m)\\
$S(R,\theta)$ --- power density (W/m$^2$)\\
$F(\theta)\!=\!S/S_{\max}$ --- normalized pattern (unitless)\\
$\beta$ --- half-power beamwidth (rad or deg)\\
$\Omega_p$ --- pattern solid angle (sr)\\
$D\!=\!4\pi/\Omega_p$ --- directivity (unitless); $\xi$ rad.\ eff.; $G\!=\!\xi D$\\
$A_e\!=\!\lambda^2 D/(4\pi)$ --- effective area (m$^2$); $A_p\!=\!l d$ phys.\ (m$^2$)\\
$R_{\text{rad}}$ --- radiation resistance ($\Omega$); $R_{\text{loss}}$ loss res.\ ($\Omega$)\\
$P_t, P_r$ --- TX / RX power (W); $G_t,G_r$ linear gains
\end{ebox}
