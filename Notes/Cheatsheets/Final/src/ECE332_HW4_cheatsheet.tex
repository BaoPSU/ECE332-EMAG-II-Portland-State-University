%-- Column 1 -----------------------------------
\shead{purplebg}{purple}{DIPOLE TYPE BY LENGTH}
\begin{ebox}
\eq{Check first}{compute $\lambda=c/f$, then $l/\lambda$}\\[1pt]
{\centering\makebox[0.98\linewidth][c]{{\setlength{\tabcolsep}{2pt}\begin{tabular}{@{}p{0.30\linewidth}p{0.30\linewidth}p{0.32\linewidth}@{}}
\toprule
$l/\lambda$ & Type & Use\\
\midrule
\rowcolor{rowB}$<1/50$ & Hertzian (short) & Hertzian formula\\
$=1/4$ & Quarter-wave & Arbitrary formula\\
\rowcolor{rowB}$=1/2$ & Half-wave & $D=1.64$, simple\\
other & Arbitrary & Arbitrary formula\\
\bottomrule
\end{tabular}}}\par}
\textit{\color{coral}If $l/\lambda<1/50$ stop and use Hertzian only --- don't subst.\ small $l$ into arbitrary formula.}
\end{ebox}

\vspace{2pt}
\shead{purplebg}{purple}{HERTZIAN DIPOLE --- $S(R,\theta)$}
\begin{ebox}
\eq{Power density}{$l/\lambda<1/50$; far field}
\[S(R,\theta) = \tfrac{\eta_0 k^2 I_0^2 l^2}{32\pi^2 R^2}\sin^2\theta\;\text{(W/m}^2\text{)}\]
\textit{$k=2\pi/\lambda$ rad/m;\enspace $I_0$ peak current (A);\enspace $l$ length (m);\enspace $R$ obs.\ dist (m).}\\
\eq{Max direction}{$\theta=\pi/2$ (broadside)}\\
\eq{Total radiated power}{integrate over $4\pi$ sr}
\[P_{\text{rad}} = \tfrac{\eta_0 \pi}{3}\!\left(\tfrac{I_0 l}{\lambda}\right)^{\!2}\;\text{(W)}\]
\eq{Directivity}{Hertzian, exact}
\[D_{\text{Hertz}} = 1.5 \quad(\text{i.e. }1.76\text{ dB})\]
\end{ebox}

\vspace{2pt}
\shead{tealbg}{teal}{ARBITRARY-LENGTH DIPOLE --- $S(\theta)$}
\begin{ebox}
\eq{Power density at distance $R$}{any $l$}
\[S(\theta) = \tfrac{15 I_0^2}{\pi R^2}\!\left[\tfrac{\cos\!\left(\tfrac{\pi l}{\lambda}\cos\theta\right)-\cos\!\left(\tfrac{\pi l}{\lambda}\right)}{\sin\theta}\right]^{\!2}\]

\eq{Normalized pattern}{$F(\theta)=S(\theta)/S_{\max}$, dimensionless}\\[1pt]

\eq{Half-wave $l=\lambda/2$}{$\pi l/\lambda=\pi/2$}
\[S(\theta) = \tfrac{15 I_0^2}{\pi R^2}\tfrac{\cos^2(\tfrac{\pi}{2}\cos\theta)}{\sin^2\theta},\quad D=1.64,\;R_{\text{rad}}=73\,\Omega\]

\eq{Quarter-wave $l=\lambda/4$}{$\pi l/\lambda=\pi/4$, $\cos(\pi/4)=\sqrt{2}/2$}
\[\tfrac{S(\theta)}{S_0}\!=\!\left[\tfrac{\cos(\tfrac{\pi}{4}\cos\theta)-\tfrac{\sqrt{2}}{2}}{\sin\theta}\right]^{\!2},\;S_0=\tfrac{15I_0^2}{\pi R^2}\]
\[S_{\max} = S(\pi/2) = S_0\!\left(1-\tfrac{\sqrt{2}}{2}\right)^{\!2}\!\approx 0.0858\,S_0\]
\end{ebox}

\columnbreak

%-- Column 2 -----------------------------------
\shead{greenbg}{green}{BEAM PARAMETERS --- HPBW, $\Omega_p$, $D$}
\begin{ebox}
\eq{1. Normalize}{}
\[F(\theta) = S(\theta)/S_{\max}\]

\eq{2. Half-power beamwidth (HPBW)}{solve $F(\theta_{1/2})=0.5$}
\[\beta = 2\theta_{1/2}\quad\text{(symmetric beams)}\]
\eq{example}{$F(\theta)=e^{-20\theta^2}$ $\Rightarrow$ $\theta_{1/2}=\sqrt{-\ln 0.5/20}=0.186$ rad $\Rightarrow$ $\beta=21.31^\circ$}\\[1pt]

\eq{3. Pattern solid angle}{(sr)}
\[\Omega_p = \!\int_0^{2\pi}\!\!\int_0^{\pi}\!F(\theta)\sin\theta\,d\theta\,d\phi\]
\textit{If $F$ has no $\phi$ dep.: $\Omega_p = 2\pi\!\int_0^\pi F(\theta)\sin\theta\,d\theta$.}

\eq{4. Directivity}{}
\[D = \tfrac{4\pi}{\Omega_p}\;\text{(unitless)} \qquad D_{\text{dB}} = 10\log_{10}D\]
\eq{5. Gain}{$\xi$ = radiation efficiency, $0<\xi\le1$}
\[G = \xi D \qquad G_{\text{dB}} = 10\log_{10}G\]
\textit{\color{coral}For lossless / ideal antennas: $G=D$.}
\end{ebox}

\vspace{2pt}
\shead{amberbg}{amber}{COMMON DIPOLE PARAMETERS}
\begin{ebox}
{\centering\makebox[0.98\linewidth][c]{{\setlength{\tabcolsep}{2pt}\begin{tabular}{@{}p{0.20\linewidth}p{0.18\linewidth}p{0.18\linewidth}p{0.30\linewidth}@{}}

\end{ebox}

\vspace{2pt}
\shead{amberbg}{amber}{EFFECTIVE AREA $A_e$}
\begin{ebox}
\eq{Definition}{$\text{m}^2$, antenna's receiving cross section}
\[A_e = \tfrac{\lambda^2 D}{4\pi}\]
\eq{Physical cross section}{half-wave dipole, wire diameter $d$}
\[A_p = l\,d = \tfrac{\lambda}{2}\,d\]
\eq{Ratio}{$A_e\gg A_p$ for thin wires (good!)}\\
\textit{Example: $f\!=\!100$ MHz, $d\!=\!2$ cm $\Rightarrow$ $A_e/A_p\!\approx\!39$.}
\end{ebox}

\vspace{2pt}
\shead{greenbg}{green}{POWER FLOW IDENTITIES}
\begin{ebox}
\[P_{\text{rad}} = S_{\max}\,\Omega_p\,R^2 = \tfrac{4\pi R^2 S_{\max}}{D}\]
\[S_{\max} = \tfrac{P_{\text{rad}}\,D}{4\pi R^2}\;\text{(W/m}^2\text{)}\]
\textit{For isotropic radiator: $S = P_{\text{rad}}/(4\pi R^2)$.}
\end{ebox}

\columnbreak

%-- Column 3 -----------------------------------
\shead{pinkbg}{pink}{FRIIS TRANSMISSION FORMULA}
\begin{ebox}
\eq{Far-field link budget}{$P_t$ tx, $P_r$ rx, dist $R$}
\[\boxed{P_r = G_t G_r\!\left(\tfrac{\lambda}{4\pi R}\right)^{\!2}\!P_t}\]
\textit{$G_t, G_r$ \emph{linear} gains;\enspace $\lambda=c/f$;\enspace $R$ in m;\enspace $P_t, P_r$ in W.}

\eq{Solve for $R$}{}
\[R = \tfrac{\lambda}{4\pi}\sqrt{\tfrac{G_tG_rP_t}{P_r}}\]

\eq{Solve for $G_r$ given $A_e$}{}
\[G_r = \tfrac{4\pi A_{e,r}}{\lambda^2}\quad\Rightarrow\quad P_r=\tfrac{G_t P_t A_{e,r}}{4\pi R^2}\]

\textit{\color{coral}Use linear $G$, not dB. Convert first: $G_{\text{lin}}=10^{G_{\text{dB}}/10}$.}\\
\textit{Free-space path loss $L_{\text{fs}}=(4\pi R/\lambda)^2$ so $P_r = G_tG_rP_t/L_{\text{fs}}$.}
\end{ebox}

\vspace{2pt}
\shead{pinkbg}{pink}{dB $\leftrightarrow$ LINEAR}
\begin{ebox}
\eq{Power gain}{factor $\to$ dB}
\[G_{\text{dB}} = 10\log_{10}G_{\text{lin}}\qquad G_{\text{lin}} = 10^{G_{\text{dB}}/10}\]

{\centering\makebox[0.98\linewidth][c]{{\setlength{\tabcolsep}{2pt}\begin{tabular}{@{}cc@{\quad}cc@{}}
\toprule
dB & linear & dB & linear\\
\midrule
\rowcolor{rowB}0 & 1 & 10 & 10\\
3 & 2 & 13 & 20\\
\rowcolor{rowB}6 & 4 & 20 & 100\\
7 & 5 & 30 & 1000\\
\rowcolor{rowB}$-3$ & $0.5$ & $-10$ & $0.1$\\
\bottomrule
\end{tabular}}}\par}
\textit{For E or amplitude use $20\log$ not $10\log$.}
\end{ebox}

\vspace{2pt}
\shead{bluebg}{blue}{RADAR EQUATION (BONUS)}
\begin{ebox}
\eq{Monostatic radar}{tx = rx, target cross section $\sigma_t$ (m$^2$)}
\[P_r = \tfrac{P_t G^2 \lambda^2 \sigma_t}{(4\pi)^3 R^4}\]
\eq{Doppler shift}{closing velocity $v_r$}
\[f_d = \tfrac{2 v_r}{\lambda} = \tfrac{2 v_r f}{c}\;\text{(Hz)}\]
\textit{Factor of 2 from round-trip path; $v_r$ is the radial component toward radar.}
\end{ebox}

\vspace{2pt}
\shead{coralbg}{coral}{ANTENNA INTEGRATION TIPS}
\begin{ebox}
\textit{Many radiation integrals (e.g.\ $\Omega_p$ for $e^{-20\theta^2}\sin\theta$) have no closed form.}\\
\textit{Use Wolfram / numerical integration. State this in the answer.}\\[1pt]
\textit{Common shortcut: if $F(\theta)$ is narrow ($\theta_{1/2}\ll1$), approximate}
\[\Omega_p \approx \theta_{1/2}^2\,\pi/(4\ln 2)\quad\text{(Gaussian beam)}\]
\end{ebox}
