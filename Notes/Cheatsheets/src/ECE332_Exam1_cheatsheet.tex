%────── COL 1: FARADAY & EMF ──────
\shead{amberbg}{amber}{MAGNETIC FLUX \& FARADAY'S LAW}
\begin{ebox}
\eq{Flux}{Wb}
\[\Phi=\iint_S\mathbf{B}\cdot d\mathbf{s}\]
\eq{Faraday's law}{$N$ turns; $N\!=\!1$ for single loop}
\[V_\text{emf}=-N\frac{d\Phi}{dt}\quad\text{(V)}\]
\textit{Lenz's law: $\mathbf{B}$ increasing $\Rightarrow\mathbf{B}_\text{ind}$ \textbf{opposes} $\mathbf{B}$;\enspace$\mathbf{B}$ decreasing $\Rightarrow\mathbf{B}_\text{ind}$ same dir.}
\end{ebox}

\vspace{0.3pt}
\shead{amberbg}{amber}{TRANSFORMER \& MOTIONAL EMF}
\begin{ebox}
\eq{Transformer EMF}{stationary loop, $\mathbf{B}$ changes}
\[V_\text{emf}^\text{tr}=-N\iint_S\frac{\partial\mathbf{B}}{\partial t}\cdot d\mathbf{s}\]
\eq{Motional EMF}{moving loop, static $\mathbf{B}$}
\[V_\text{emf}^\text{m}=\oint_C(\mathbf{u}\times\mathbf{B})\cdot d\mathbf{l}\]
\eq{Total (general)}{both loop moves and $\mathbf{B}$ varies}
\[V_\text{emf}=-\frac{d\Phi}{dt}=V_\text{emf}^\text{tr}+V_\text{emf}^\text{m}\]
\textit{If $\mathbf{u}\!\parallel\!d\mathbf{l}$ or $\mathbf{B}\!\parallel\!d\mathbf{l}$: that segment contributes 0 to $V_\text{emf}^\text{m}$.}
\end{ebox}

\vspace{0.3pt}
\shead{amberbg}{amber}{AC GENERATOR}
\begin{ebox}
\textit{Loop area $A$, $N$ turns, angular vel.\ $\omega$, initial angle $C_0$, field $B_0$:}
\[\Phi=AB_0\cos(\omega t+C_0),\qquad V_\text{emf}=NA\omega B_0\sin(\omega t+C_0)\]
\textit{Peak $V_0=NA\omega B_0$.\enspace To increase: $\uparrow N,\,A,\,\omega,\,B_0$.}
\end{ebox}

\vspace{0.3pt}
\shead{tealbg}{teal}{IDEAL TRANSFORMER}
\begin{ebox}
\[\frac{V_1}{V_2}=\frac{N_1}{N_2},\qquad\frac{I_1}{I_2}=\frac{N_2}{N_1}\]
\[Z_\text{in}=\!\left(\!\frac{N_1}{N_2}\!\right)^{\!2}\!Z_L,\quad R_\text{in}=\!\left(\!\frac{N_1}{N_2}\!\right)^{\!2}\!R_L\]
\textit{Inductor in changing $\mathbf{B}$: $I=V_\text{emf}^\text{tr}/(R+R_i)\approx V_\text{emf}^\text{tr}/R$ when $R_i\!\ll\!R$.}
\end{ebox}

\shead{bluebg}{blue}{MAXWELL'S EQUATIONS}
\begin{ebox}

\vspace{0.3pt}
\shead{bluebg}{blue}{DISPLACEMENT CURRENT}
\begin{ebox}
\[\mathbf{J}_d=\frac{\partial\mathbf{D}}{\partial t}=\varepsilon\frac{\partial\mathbf{E}}{\partial t}\;\text{(A/m}^2\text{)},\qquad I_d=\iint_S\mathbf{J}_d\cdot d\mathbf{s}\;\text{(A)}\]
\textit{Phasor: $\tilde{\mathbf{J}}_d=j\omega\varepsilon\tilde{\mathbf{E}}$.\enspace Makes total current $I=I_c+I_d$ continuous.}
\end{ebox}

\vspace{0.3pt}
\shead{coralbg}{coral}{CHARGE CONTINUITY \& RELAXATION}
\begin{ebox}
\[\nabla\cdot\mathbf{J}=-\frac{\partial\rho_v}{\partial t}\;\Longrightarrow\;\oint_S\mathbf{J}\cdot d\mathbf{s}=0\;\text{(KCL)}\]
\[\tau_r=\frac{\varepsilon}{\sigma}\;\text{(s)},\qquad\rho_v(t)=\rho_{v0}\,e^{-t/\tau_r}\;\text{(C/m}^3\text{)}\]
\textit{Cu: $\tau_r\!\approx\!10^{-19}$\,s;\enspace mica: $\tau_r\!\approx\!10^{4}$\,s.}
\end{ebox}

\vspace{0.3pt}
\shead{bluebg}{blue}{DYNAMIC EM POTENTIALS}
\begin{ebox}
\[\mathbf{B}=\nabla\times\mathbf{A},\qquad\mathbf{E}=-\nabla V-\frac{\partial\mathbf{A}}{\partial t}\]
\eq{Retarded potentials}{effects propagate at speed $u_p$, $R'=|\mathbf{R}-\mathbf{R}_i|$}
\[V(\mathbf{R},t)=\frac{1}{4\pi\varepsilon}\int\frac{\rho_v(\mathbf{R}_i,\,t-R'/u_p)}{R'}\,d\mathcal{V}'\]
\[\mathbf{A}(\mathbf{R},t)=\frac{\mu}{4\pi}\int\frac{\mathbf{J}(\mathbf{R}_i,\,t-R'/u_p)}{R'}\,d\mathcal{V}'\]
\end{ebox}

\vspace{0.3pt}
\shead{pinkbg}{pink}{TIME-HARMONIC POTENTIALS \& PHASORS}
\begin{ebox}
\textit{$k=\omega/u_p$ (wavenumber);\enspace $e^{j\omega t}$ factor stripped to get phasor.}
\[\tilde{V}(\mathbf{R})=\frac{1}{4\pi\varepsilon}\int\frac{\tilde{\rho}_v(\mathbf{R}_i)e^{-jkR'}}{R'}\,d\mathcal{V}'\]
\[\tilde{\mathbf{H}}=\frac{1}{\mu}\nabla\times\tilde{\mathbf{A}},\qquad\tilde{\mathbf{A}}=\frac{\mu}{4\pi}\int\frac{\tilde{\mathbf{J}}(\mathbf{R}_i)e^{-jkR'}}{R'}\,d\mathcal{V}'\]
\textit{Given $\tilde{\mathbf{E}}$: apply $\tilde{\mathbf{H}}=-\frac{1}{j\omega\mu}\nabla\times\tilde{\mathbf{E}}$ (Faraday in phasor domain).}
\end{ebox}

%────── COL 3: LOSSLESS PLANE WAVE ──────
\shead{purplebg}{purple}{LOSSLESS PLANE WAVE}
\begin{ebox}
\eq{Wave equation (phasor, source-free, $\sigma=0$)}{}
\[\nabla^2\tilde{\mathbf{E}}+k^2\tilde{\mathbf{E}}=0,\qquad k=\omega\sqrt{\mu\varepsilon}\;\text{(rad/m)}\]
\eq{Intrinsic impedance}{}
\[\eta=\sqrt{\frac{\mu}{\varepsilon}}\;(\Omega),\qquad\eta_0=\sqrt{\frac{\mu_0}{\varepsilon_0}}\approx377\;\Omega\approx120\pi\;\Omega\]
\eq{Phase velocity \& wavelength}{}
\[u_p=\frac{1}{\sqrt{\mu\varepsilon}}=\frac{\omega}{k}\;\text{(m/s)},\qquad\lambda=\frac{2\pi}{k}=\frac{u_p}{f}\;\text{(m)}\]
\eq{Fields (wave in $+\hat{z}$, $\mathbf{E}$ in $\hat{x}$)}{}
\[\tilde{\mathbf{E}}(z)=E_{x0}^+e^{-jkz}\hat{x},\qquad\tilde{\mathbf{H}}(z)=\frac{E_{x0}^+}{\eta}e^{-jkz}\hat{y}\]
\[\mathbf{E}(z,t)=|E_{x0}^+|\cos(\omega t-kz+\phi^+)\hat{x}\]
\[\mathbf{H}(z,t)=\frac{|E_{x0}^+|}{\eta}\cos(\omega t-kz+\phi^+)\hat{y}\]
\textit{TEM: $\tilde{\mathbf{E}}\perp\tilde{\mathbf{H}}\perp\hat{k}$.\enspace In lossless medium $\mathbf{E}$ and $\mathbf{H}$ are \textbf{in phase}.}
\end{ebox}

\vspace{0.3pt}
\shead{purplebg}{purple}{DIRECTION RELATIONS}
\begin{ebox}
\[\tilde{\mathbf{H}}=\frac{1}{\eta}\hat{k}\times\tilde{\mathbf{E}},\qquad\tilde{\mathbf{E}}=-\eta\,\hat{k}\times\tilde{\mathbf{H}}\]

\bottomrule
\end{tabular}}\par}
\end{ebox}

\vspace{0.3pt}
\shead{pinkbg}{pink}{PHASOR $\leftrightarrow$ TIME DOMAIN}
\begin{ebox}

\textit{Back to time: $\mathbf{E}(z,t)=\operatorname{Re}\!\left\{\tilde{\mathbf{E}}(z)e^{j\omega t}\right\}$}
\end{ebox}

%────── COL 1: POLARIZATION ──────
\shead{tealbg}{teal}{POLARIZATION}
\begin{ebox}
\textit{Polarization = curve traced by $\mathbf{E}$ in plane $\perp\hat{k}$ as $t$ varies.}\\[0.3pt]
\eq{General phasor (wave in $+\hat{z}$)}{$a_x\!=\!|E_{x0}|$, $a_y\!=\!|E_{y0}|$, $\delta$ = phase diff $E_y$ rel.\ $E_x$}
\[\tilde{\mathbf{E}}(z)=\bigl(a_x\hat{x}+a_ye^{j\delta}\hat{y}\bigr)e^{-jkz}\]
\[\mathbf{E}(z,t)=a_x\cos(\omega t-kz)\hat{x}+a_y\cos(\omega t-kz+\delta)\hat{y}\]
\end{ebox}

\vspace{0.3pt}
\shead{tealbg}{teal}{LINEAR POLARIZATION}
\begin{ebox}
\textit{Condition: $\delta=0$ or $\delta=\pi$. $E_x$, $E_y$ in/out of phase $\Rightarrow$ straight line.}
\[\mathbf{E}=(a_x\hat{x}\pm a_y\hat{y})\cos(\omega t-kz),\quad|\mathbf{E}|=\sqrt{a_x^2+a_y^2}\,|\cos(\cdot)|\]
\[\psi=\tan^{-1}\!\left(\frac{\pm a_y}{a_x}\right)\text{(tilt from $+\hat{x}$)}\]
\textit{$a_y\!=\!0$: x-pol ($\psi\!=\!0$);\enspace$a_x\!=\!0$: y-pol ($\psi\!=\!\pm\pi/2$).}
\end{ebox}

\vspace{0.3pt}
\shead{tealbg}{teal}{CIRCULAR POLARIZATION}
\begin{ebox}
\textit{Condition: $a_x=a_y=a>0$, $\delta=\pm\pi/2$.\enspace $|\mathbf{E}|=a$ = const.}\\[0.3pt]
\eq{LHC ($\delta=+\pi/2$)}{}
\[\tilde{\mathbf{E}}=a(\hat{x}+j\hat{y})e^{-jkz},\quad\mathbf{E}=a\bigl[\cos(\omega t-kz)\hat{x}-\sin(\omega t-kz)\hat{y}\bigr]\]
\eq{RHC ($\delta=-\pi/2$)}{}
\[\tilde{\mathbf{E}}=a(\hat{x}-j\hat{y})e^{-jkz},\quad\mathbf{E}=a\bigl[\cos(\omega t-kz)\hat{x}+\sin(\omega t-kz)\hat{y}\bigr]\]
\end{ebox}

\vspace{0.3pt}
\shead{tealbg}{teal}{ELLIPTICAL POLARIZATION}
\begin{ebox}
\textit{General ($a_x\ne a_y$, $\delta\ne0,\pm\pi/2,\pi$): $\mathbf{E}$ traces an ellipse.}
\[\psi_0=\tan^{-1}\!\!\left(\frac{a_y}{a_x}\right)\!\in\!\left[0,\frac{\pi}{2}\right]\;\text{(auxiliary angle)}\]
\[\tan 2\gamma=(\tan 2\psi_0)\cos\delta\quad\text{(rotation angle $\gamma$)}\]
\[\sin 2\chi=(\sin 2\psi_0)\sin\delta\quad\text{(ellipticity angle $\chi$)}\]
\[R=\frac{a_\xi}{a_\eta}\;\text{(axial ratio)},\qquad\tan\chi=\pm\frac{1}{R}\]
\textit{$\tan\chi>0$: LH;\enspace$\tan\chi<0$: RH;\enspace$R=1$: circular;\enspace$a_\eta=0$: linear.}
\end{ebox}

%────── COL 2: LOSSY MEDIUM ──────
\shead{coralbg}{coral}{EM WAVES IN LOSSY MEDIUM}
\begin{ebox}
\eq{Complex permittivity}{}
\[\varepsilon_c=\varepsilon'-j\varepsilon''=\varepsilon_r\varepsilon_0-j\frac{\sigma}{\omega},\qquad\frac{\varepsilon''}{\varepsilon'}=\frac{\sigma}{\omega\varepsilon'}\]
\eq{Propagation constant $\gamma=\alpha+j\beta$;\quad$\gamma^2=-\omega^2\mu\varepsilon_c$}{}
\[\alpha=\omega\!\sqrt{\frac{\mu\varepsilon'}{2}}\!\left[\sqrt{1\!+\!\left(\!\frac{\varepsilon''}{\varepsilon'}\!\right)^{\!2}}-1\right]^{\!\frac{1}{2}}\!\!\text{(Np/m)}\]
\[\beta=\omega\!\sqrt{\frac{\mu\varepsilon'}{2}}\!\left[\sqrt{1\!+\!\left(\!\frac{\varepsilon''}{\varepsilon'}\!\right)^{\!2}}+1\right]^{\!\frac{1}{2}}\!\!\text{(rad/m)}\]
\eq{Intrinsic impedance}{$\eta_c=|\eta_c|e^{j\theta_\eta}$, \textbf{complex}}
\[\eta_c=\sqrt{\frac{\mu}{\varepsilon_c}}=\sqrt{\frac{\mu}{\varepsilon'}}\!\left(1-j\frac{\varepsilon''}{\varepsilon'}\right)^{\!-1/2}\]
\eq{Fields (wave in $+\hat{z}$, $\mathbf{E}\!\parallel\!\hat{x}$)}{}
\[\tilde{\mathbf{E}}(z)=E_{x0}e^{-\alpha z}e^{-j\beta z}\hat{x},\quad\tilde{\mathbf{H}}(z)=\frac{E_{x0}}{\eta_c}e^{-\alpha z}e^{-j\beta z}\hat{y}\]
\textit{$\mathbf{E}$, $\mathbf{H}$ \textbf{not in phase} in lossy media (phase diff $\theta_\eta$).}
\end{ebox}

\vspace{0.3pt}
\shead{coralbg}{coral}{MEDIUM CLASSIFICATION}
\begin{ebox}
{\centering\makebox[0.98\linewidth][c]{{\setlength{\tabcolsep}{2pt}\begin{tabular}{@{}p{0.25\linewidth}p{0.16\linewidth}p{0.51\linewidth}@{}}
\toprule
Type & $\varepsilon''/\varepsilon'$ & Approximations\\
\midrule
\rowcolor{rowB}Low-loss diel. & $\ll1$ & $\alpha\!\approx\!\tfrac{\sigma}{2}\!\sqrt{\tfrac{\mu}{\varepsilon'}}$,\;$\beta\!\approx\!\omega\sqrt{\mu\varepsilon'}$,\;$\eta_c\!\approx\!\sqrt{\mu/\varepsilon'}$\\
Quasi-cond. & $10^{-2}\!-\!10^2$ & Use general $\alpha$, $\beta$\\
\rowcolor{rowB}Good cond. & $\gg100$ & $\alpha\!\approx\!\beta\!\approx\!\sqrt{\pi f\mu\sigma}$,\;$\eta_c\!\approx\!(1\!+\!j)\!\sqrt{\pi f\mu/\sigma}$\\
\bottomrule
\end{tabular}}}\par}
\eq{Skin depth}{$|\mathbf{E}|$ drops to $1/e$ of surface value}
\[\delta_s=\frac{1}{\alpha}\;\text{(m)},\qquad\delta_s^\text{cond}=\frac{1}{\sqrt{\pi f\mu\sigma}}\;\text{(m)}\]
\textit{$|\tilde{\mathbf{E}}(z)|=|E_{x0}|e^{-\alpha z}$;\enspace$|\tilde{\mathbf{H}}(z)|=\frac{|E_{x0}|}{|\eta_c|}e^{-\alpha z}$.}
\end{ebox}

\vspace{0.3pt}
\shead{coralbg}{coral}{CURRENT IN GOOD CONDUCTOR}
\begin{ebox}
\textit{AC current flows near surface; DC current uniform throughout conductor.}
\eq{Current density at depth $z$}{$J_0=\sigma E_0$, $E_0$ at surface}
\[\tilde{\mathbf{J}}(z)=J_0\,e^{-(1+j)z/\delta_s}\hat{x}\;\text{(A/m}^2\text{)}\]
\eq{Total current, slab width $w$ ($y$-dir)}{}
\[\tilde{I}=\frac{J_0 w\delta_s}{1+j}\;\text{(A)}\]
\eq{Surface impedance}{}
\[Z_s=\frac{1+j}{\sigma\delta_s}=R_s(1+j),\qquad R_s=\sqrt{\frac{\pi f\mu}{\sigma}}\;(\Omega)\]
\eq{AC resistance: slab (width $w$, length $\ell$)}{}
\[R=R_s\frac{\ell}{w}\;(\Omega),\qquad R'_\text{coax}=\frac{R_s}{2\pi}\!\left(\frac{1}{a}+\frac{1}{b}\right)\;(\Omega/\text{m})\]
\end{ebox}

%────── COL 3: POWER + dB ──────
\shead{greenbg}{green}{EM POWER DENSITY (POYNTING)}
\begin{ebox}
\eq{Instantaneous Poynting vector}{W/m$^2$}
\[\mathbf{S}=\mathbf{E}\times\mathbf{H}\;\text{(W/m}^2\text{)}\]
\eq{Average power density (time-harmonic)}{}
\[\mathbf{S}_\text{av}=\frac{1}{2}\operatorname{Re}\!\left\{\tilde{\mathbf{E}}\times\tilde{\mathbf{H}}^*\right\}\;\text{(W/m}^2\text{)}\]
\eq{Lossless ($\alpha=0$, $\theta_\eta=0$, $\eta$ real)}{}
\[\mathbf{S}_\text{av}=\frac{|E_0|^2}{2\eta}\hat{k}\;\text{(W/m}^2\text{)}\]
\eq{Lossy ($\eta_c=|\eta_c|e^{j\theta_\eta}$, wave in $+\hat{z}$)}{}
\[\mathbf{S}_\text{av}=\frac{|\tilde{E}(0)|^2}{2|\eta_c|}\,e^{-2\alpha z}\cos\theta_\eta\;\hat{z}\;\text{(W/m}^2\text{)}\]
\textit{Power $\propto e^{-2\alpha z}$, field $\propto e^{-\alpha z}$: power decays at \textbf{2$\times$} the rate.}\\
\eq{Power through area $A$, angle $\theta$ to $\hat{n}$}{}
\[P=S_\text{av}A\cos\theta\;\text{(W)}\]
\end{ebox}

\vspace{0.3pt}
\shead{graybg}{gray}{ATTENUATION \& DECIBELS}
\begin{ebox}
\eq{Unit conversion}{}
\[\alpha\;\text{(dB/m)}=8.69\,\alpha\;\text{(Np/m)}\]
\eq{Power attenuation rate from $z=0$}{}
\[A=10\log\!\left(\frac{S_\text{av}(z)}{S_\text{av}(0)}\right)=-8.69\,\alpha z\;\text{(dB)}\]
\eq{Field attenuation from $z=0$}{}
\[A=20\log\!\left(\frac{|E(z)|}{|E(0)|}\right)\;\text{(dB)}\]
\eq{Power/voltage ratio}{}
\[G=10\log\!\left(\frac{P_1}{P_2}\right)=20\log\!\left(\frac{V_1}{V_2}\right)\;\text{(dB)}\]
\textit{$P\!\times\!2\Rightarrow\!+3$\,dB;\enspace$P\!\times\!10\Rightarrow\!+10$\,dB;\enspace$|E|\!\times\!2\Rightarrow\!+6$\,dB.}
\end{ebox}

\vspace{0.3pt}
\shead{graybg}{gray}{READ-A-WAVE WORKFLOW}
\begin{ebox}
\textbf{Given $\mathbf{E}(z,t)=E_0\cos(\omega t-kz+\phi)\hat{a}$:}\\
1. Read: $\omega$, $k$, $E_0$, $\phi$, dir.\ $\hat{a}$, prop.\ dir.\ $\hat{k}$\\
2. $f=\omega/2\pi$;\enspace$\lambda=2\pi/k$;\enspace$u_p=\omega/k$\\
3. $\eta=\sqrt{\mu/\varepsilon}$ (or $\mu u_p$); free space: $\eta_0\approx377\,\Omega$\\
