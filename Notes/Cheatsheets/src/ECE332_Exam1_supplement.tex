\shead{purplebg}{purple}{EMF \& FARADAY'S LAW}
\begin{ebox}
\eq{Transformer EMF}{stationary loop, $\mathbf{B}$ changes}
\[V_\text{emf}=-\iint_S\frac{\partial\mathbf{B}}{\partial t}\cdot d\mathbf{s}\]
\eq{Motional EMF}{loop moves, $\mathbf{B}$ static}
\[V_\text{emf}=\oint(\mathbf{v}\times\mathbf{B})\cdot d\mathbf{l}\]
\eq{General (flux form)}{always works}
\[V_\text{emf}=-\frac{d\Phi}{dt},\qquad\Phi=\iint_S\mathbf{B}\cdot d\mathbf{s}\]
\eq{$N$-turn coil}{}
\[V_\text{emf}=-N\frac{d\Phi}{dt}=-\frac{d\lambda}{dt},\qquad\lambda=N\Phi\]
\eq{Uniform $\mathbf{B}$ shortcut}{$\mathbf{B}$ same everywhere on loop}
\[V_\text{emf}=-\frac{\partial B}{\partial t}\cdot A\]
\eq{Long wire + rect loop}{wire near loop, $r$: $a\to b$, height $c$}
\[V_\text{emf}=-\int_0^c\!\int_a^b\frac{\partial}{\partial t}\!\left(\frac{\mu_0 I}{2\pi r}\right)dr\,dz\]
\end{ebox}

\vspace{0.3pt}
\shead{amberbg}{amber}{ROTATING LOOP — GENERATOR}
\begin{ebox}
\[\Phi=AB_0\cos(\omega t+\phi_0),\qquad V_\text{emf}=AB_0\omega\sin(\omega t+\phi_0)\]
\[V_0=NA\omega B_0\;\text{(peak)},\qquad\omega=2\pi f=2\pi\frac{\text{rpm}}{60}\]
\textit{2nd method: $V_\text{emf}=\oint(\mathbf{u}\times\mathbf{B})\cdot d\mathbf{l}$ on moving sides only.}
\end{ebox}

\vspace{0.3pt}
\shead{tealbg}{teal}{IDEAL TRANSFORMER}
\begin{ebox}
\[\frac{V_1}{V_2}=\frac{N_1}{N_2},\qquad\frac{I_1}{I_2}=\frac{N_2}{N_1},\qquad Z_\text{in}=\!\left(\!\frac{N_1}{N_2}\!\right)^{\!2}\!Z_L\]
\textit{Step-up ($N_2>N_1$): $V_2>V_1$, $I_2<I_1$. Power conserved: $V_1I_1=V_2I_2$.}

\shead{coralbg}{coral}{CONDUCTION vs DISPLACEMENT CURRENT}
\begin{ebox}
\eq{Conduction}{resistive, $\sigma\ne0$}
\[E=\frac{V(t)}{d},\quad J_c=\sigma E,\quad I_c=\frac{\sigma A}{d}V(t)=\frac{V}{R},\quad R=\frac{d}{\sigma A}\]

\eq{Lossy capacitor ($R\parallel C$)}{both $\sigma$ and $\varepsilon_r$}
\[I=I_c+I_d=\frac{V}{R}+C\frac{dV}{dt}\]
\textit{Phasor: $\tilde{J}_d=j\omega\varepsilon\tilde{E}$.\enspace In good conductor $I_d\ll I_c$.}

%────── COL 1: EXACT LOSSY MEDIUM FORMULAS ──────
\shead{coralbg}{coral}{LOSSY MEDIUM: EXACT GENERAL FORMULAS $\bigstar$}
\begin{ebox}
\eq{Complex permittivity}{}
\[\varepsilon_c=\varepsilon'-j\varepsilon''=\varepsilon_r\varepsilon_0-j\frac{\sigma}{\omega},\qquad\varepsilon''=\frac{\sigma}{\omega}\]
\eq{Propagation constant}{$\gamma^2=-\omega^2\mu\varepsilon_c$}
\[\gamma=\alpha+j\beta\]
\eq{Exact $\alpha$ (attenuation)}{Np/m}
\[\alpha=\omega\!\sqrt{\frac{\mu\varepsilon'}{2}}\!\left[\sqrt{1\!+\!\left(\frac{\varepsilon''}{\varepsilon'}\right)^{\!2}}-1\right]^{\!\frac{1}{2}}\]
\eq{Exact $\beta$ (phase constant)}{rad/m}
\[\beta=\omega\!\sqrt{\frac{\mu\varepsilon'}{2}}\!\left[\sqrt{1\!+\!\left(\frac{\varepsilon''}{\varepsilon'}\right)^{\!2}}+1\right]^{\!\frac{1}{2}}\]
\eq{Complex intrinsic impedance}{}
\[\eta_c=\sqrt{\frac{\mu}{\varepsilon_c}}=|\eta_c|e^{j\theta_\eta},\qquad\theta_\eta\in\left[0,\frac{\pi}{4}\right]\]
\eq{Lossy phasor fields}{wave in $+\hat{z}$, $\mathbf{E}\!\parallel\!\hat{x}$}
\[\tilde{\mathbf{E}}(z)=E_0\,e^{-\alpha z}e^{-j\beta z}\hat{x},\quad\tilde{\mathbf{H}}(z)=\frac{E_0}{\eta_c}\,e^{-\alpha z}e^{-j\beta z}\hat{y}\]
\textit{$\mathbf{E}$ and $\mathbf{H}$ \textbf{out of phase} by $\theta_\eta$ in lossy medium (unlike lossless).}\\[0.5pt]
\textit{Real fields: $\mathbf{H}(z,t)=\frac{|E_0|}{|\eta_c|}e^{-\alpha z}\cos(\omega t-\beta z-\theta_\eta)\hat{y}$}
\end{ebox}

%────── COL 2: CURRENT IN GOOD CONDUCTOR ──────
\shead{coralbg}{coral}{CURRENT IN GOOD CONDUCTOR $\bigstar$}
\begin{ebox}
\textit{AC current flows near surface in layer $\sim\delta_s$; DC current uniform.}\\[0.5pt]
\eq{Current density at depth $z$}{$J_0=\sigma E_0$, surface at $z=0$}
\[\tilde{\mathbf{J}}(z)=J_0\,e^{-(1+j)z/\delta_s}\hat{x}\;\text{(A/m}^2\text{)}\]
\eq{Total current, slab width $w$ ($y$-dir, integrate $z$: $0\to\infty$)}{}
\[\tilde{I}=\frac{J_0\,w\,\delta_s}{1+j}\;\text{(A)}\]
\eq{Surface impedance}{per square}
\[Z_s=\frac{1+j}{\sigma\delta_s}=R_s(1+j),\qquad R_s=\sqrt{\frac{\pi f\mu}{\sigma}}\;(\Omega)\]
\eq{AC resistance, slab (width $w$, length $\ell$)}{}
\[R_\text{ac}=R_s\frac{\ell}{w}\;(\Omega)\]
\eq{Coaxial AC resistance}{inner $a$, outer $b$}
\[R'_\text{coax}=\frac{R_s}{2\pi}\!\left(\frac{1}{a}+\frac{1}{b}\right)\;(\Omega/\text{m})\]
\textit{Power dissipated: $P=\tfrac{1}{2}|I|^2 R_s(\ell/w)$.}\\[0.3pt]
\textit{$R_s\propto\sqrt{f}$: higher frequency $\Rightarrow$ more loss, smaller $\delta_s$.}
\end{ebox}
