%────── COL 1: NORMAL INCIDENCE ──────
\shead{coralbg}{coral}{NORMAL INCIDENCE (boundary at $z=0$)}
\begin{ebox}
\textit{Medium 1 ($\eta_1,k_1$) for $z\!<\!0$;\enspace Medium 2 ($\eta_2,k_2$) for $z\!\geq\!0$.}\\[0.5pt]
\eq{Reflection \& transmission coefficients}{}
\[\Gamma=\frac{\eta_2-\eta_1}{\eta_2+\eta_1},\qquad\tau=\frac{2\eta_2}{\eta_2+\eta_1}=1+\Gamma\]
\eq{Power conservation}{}
\[|\Gamma|^2+\frac{\eta_1}{\eta_2}|\tau|^2=1\]
\eq{Fields in medium 1}{incident + reflected}
\[\tilde{\mathbf{E}}_1=E_0^+(e^{-jk_1z}+\Gamma e^{+jk_1z})\hat{x}\]
\[\tilde{\mathbf{H}}_1=\frac{E_0^+}{\eta_1}(e^{-jk_1z}-\Gamma e^{+jk_1z})\hat{y}\quad\textit{(note $-\Gamma$)}\]
\eq{Fields in medium 2}{transmitted only}
\[\tilde{\mathbf{E}}_2=\tau E_0^+e^{-jk_2z}\hat{x},\qquad\tilde{\mathbf{H}}_2=\frac{\tau E_0^+}{\eta_2}e^{-jk_2z}\hat{y}\]
\eq{Special cases}{}
{\centering\makebox[0.98\linewidth][c]{{\setlength{\tabcolsep}{2pt}\begin{tabular}{@{}p{0.32\linewidth}p{0.16\linewidth}p{0.10\linewidth}p{0.10\linewidth}@{}}
\toprule
Case & $\eta_2$ & $\Gamma$ & $\tau$\\
\midrule
\rowcolor{rowB}PEC (perfect cond.) & $0$ & $-1$ & $0$\\
PMC & $\infty$ & $+1$ & $2$\\
\rowcolor{rowB}Matched & $\eta_1$ & $0$ & $1$\\
\bottomrule
\end{tabular}}}\par}
\end{ebox}

\vspace{0.3pt}
\shead{coralbg}{coral}{STANDING WAVES \& VSWR}
\begin{ebox}
\eq{VSWR (voltage standing-wave ratio)}{}
\[S=\frac{1+|\Gamma|}{1-|\Gamma|}\in[1,\infty)\quad\Leftrightarrow\quad|\Gamma|=\frac{S-1}{S+1}\]
\eq{Envelope in medium 1}{}
\[|\tilde{E}_1|_\text{max}=|E_0^+|(1+|\Gamma|),\quad|\tilde{E}_1|_\text{min}=|E_0^+|(1-|\Gamma|)\]
\eq{PEC ($\Gamma\!=\!-1$) standing wave — zero $\hat{k}$ component}{}
\[|\tilde{\mathbf{E}}_1(z)|=2|E_0^+||\sin(k_1 z)|\]
\textit{Zeros at $z=0,-\lambda_1/2,-\lambda_1,\ldots$\enspace (multiples of $\lambda/2$)}\\
\textit{Maxima at $z=-\lambda_1/4,-3\lambda_1/4,\ldots$}\\[0.5pt]
\eq{Net power flow in medium 1}{}
\[S_{1,\text{av}}=\frac{|E_0^+|^2}{2\eta_1}(1-|\Gamma|^2)\;\hat{z}\]
\textit{PEC: zero net power ($|\Gamma|=1$).}
\end{ebox}

\columnbreak

%────── COL 2: OBLIQUE INCIDENCE ──────
\shead{bluebg}{blue}{OBLIQUE INCIDENCE}
\begin{ebox}
\textit{$\theta_i$=incidence angle (from normal); $\theta_r=\theta_i$; $\theta_t$=transmitted.}\\[0.5pt]
\eq{Snell's law}{$n=\sqrt{\mu_r\varepsilon_r}$}
\[n_1\sin\theta_i=n_2\sin\theta_t,\qquad n=\frac{c}{u_p}=\sqrt{\mu_r\varepsilon_r}\]
\textit{Non-magnetic ($\mu_{r1}\!=\!\mu_{r2}\!=\!1$): $\sqrt{\varepsilon_1}\sin\theta_i=\sqrt{\varepsilon_2}\sin\theta_t$.}\\[0.5pt]
\eq{TE polarization}{$\mathbf{E}\perp$ plane of incidence ("$s$")}
\[\Gamma_\text{TE}=\frac{\eta_2\cos\theta_i-\eta_1\cos\theta_t}{\eta_2\cos\theta_i+\eta_1\cos\theta_t}\]
\[\tau_\text{TE}=\frac{2\eta_2\cos\theta_i}{\eta_2\cos\theta_i+\eta_1\cos\theta_t}\]
\eq{TM polarization}{$\mathbf{H}\perp$ plane of incidence ("$p$")}
\[\Gamma_\text{TM}=\frac{\eta_2\cos\theta_t-\eta_1\cos\theta_i}{\eta_2\cos\theta_t+\eta_1\cos\theta_i}\]
\[\tau_\text{TM}=\frac{2\eta_2\cos\theta_i}{\eta_2\cos\theta_t+\eta_1\cos\theta_i}\]
\eq{Power reflectance \& transmittance}{}
\[R=|\Gamma|^2,\qquad T=1-R=\frac{\eta_1\cos\theta_t}{\eta_2\cos\theta_i}|\tau|^2\]
\end{ebox}

\vspace{0.3pt}
\shead{bluebg}{blue}{BREWSTER'S \& CRITICAL ANGLES}
\begin{ebox}
\eq{Brewster's angle}{$\Gamma_\text{TM}=0$; non-magnetic only}
\[\tan\theta_B=\frac{n_2}{n_1}=\sqrt{\frac{\varepsilon_2}{\varepsilon_1}}\quad\text{(no TE Brewster's)}\]
\textit{At $\theta_B$ TM polarization is \textbf{fully transmitted}; TE still partly reflected.}\\[0.5pt]
\eq{Critical angle}{TIR; requires $n_1>n_2$}
\[\sin\theta_c=\frac{n_2}{n_1},\qquad\theta_c=\arcsin\!\left(\frac{n_2}{n_1}\right)\]
\textit{For $\theta_i\geq\theta_c$: $|\Gamma|=1$ (both TE and TM), evanescent wave in medium 2.}\\[0.5pt]
\eq{Optical fiber (step-index)}{$n_\text{core}>n_\text{clad}$}
\[\text{NA}=n_0\sin\theta_a=\sqrt{n_\text{core}^2-n_\text{clad}^2}\]
\textit{$\theta_a$ = acceptance half-angle (outside). $n_0=1$ in air.}
\end{ebox}

\columnbreak

%────── COL 3: MULTI-LAYER / WAVE IMPEDANCE ──────
\shead{purplebg}{purple}{WAVE IMPEDANCE \& MULTI-LAYER}
\begin{ebox}
\textit{Input impedance at $z=-d$ looking toward load $Z_L$ at $z=0$, medium $\eta$:}
\[Z_\text{in}=\eta\,\frac{Z_L+j\eta\tan(\beta d)}{\eta+jZ_L\tan(\beta d)}\]
\eq{Quarter-wave transformer ($d=\lambda/4$, $\tan\to\infty$)}{}
\[Z_\text{in}=\frac{\eta^2}{Z_L}\quad\Rightarrow\quad\eta=\sqrt{Z_\text{in}\cdot Z_L}\;\text{for match}\]
\eq{Half-wave ($d=\lambda/2$)}{transparent}
\[Z_\text{in}=Z_L\]
\eq{PEC load ($Z_L=0$)}{short circuit}
\[Z_\text{in}=j\eta\tan(\beta d)\quad\text{(pure reactance)}\]
\eq{3-layer anti-reflection coating}{$\lambda/4$ slab, medium $\eta_2$}
\[\Gamma=0\;\Leftrightarrow\;\eta_2=\sqrt{\eta_1\eta_3},\;d=\lambda_2/4\]
\end{ebox}

\vspace{0.3pt}
\shead{graybg}{gray}{USEFUL TRIG FOR OBLIQUE}
\begin{ebox}
\textit{Find $\cos\theta_t$ after Snell's:}
\[\cos\theta_t=\sqrt{1-\sin^2\theta_t}=\sqrt{1-\frac{n_1^2}{n_2^2}\sin^2\theta_i}\]
\textit{Complex if $\theta_i>\theta_c$ (TIR — $\cos\theta_t$ becomes imaginary).}\\[0.5pt]
\textit{Check: for normal incidence ($\theta_i=0$): $\cos\theta_i=\cos\theta_t=1$; Fresnel $\to$ normal-incidence $\Gamma$.}
\end{ebox}

%────── COL 1: RADIATION & SHORT DIPOLE ──────
\shead{amberbg}{amber}{RADIATION FUNDAMENTALS}
\begin{ebox}
\textit{Far field ($r\!\gg\!\lambda$): only $\hat{\theta}$ and $\hat{\phi}$ components survive. $\hat{r}$ component decays as $1/r^2$.}\\[0.5pt]
\eq{Radiation intensity}{W/sr; $r^2$ times power density}
\[U(\theta,\phi)=\frac{r^2}{2\eta_0}|\tilde{E}_\theta|^2\;\text{(W/sr)}\]
\eq{Total radiated power}{}
\[P_\text{rad}=\iint_\Omega U\,d\Omega=\int_0^{2\pi}\!\!\int_0^\pi U\sin\theta\,d\theta\,d\phi\]
\eq{Normalized pattern}{$F_\text{max}=1$}
\[F(\theta,\phi)=\frac{U(\theta,\phi)}{U_\text{max}}\]
\end{ebox}

\vspace{0.3pt}
\shead{amberbg}{amber}{HERTZIAN (SHORT) DIPOLE}
\begin{ebox}
\textit{Length $dl\!\ll\!\lambda$, current $I_0$, at origin, oriented $\hat{z}$:}\\[0.5pt]
\eq{Far-field phasors}{}
\[\tilde{E}_\theta=j\frac{\eta_0 I_0\,dl\sin\theta}{2\lambda\,r}\,e^{-jkr},\qquad\tilde{H}_\phi=\frac{\tilde{E}_\theta}{\eta_0}\]
\eq{Radiation intensity}{}
\[U(\theta)=\frac{\eta_0}{8}\!\left(\frac{I_0\,dl}{\lambda}\right)^{\!2}\!\sin^2\theta,\qquad F(\theta)=\sin^2\theta\]
\eq{Total radiated power}{}
\[P_\text{rad}=\frac{\pi}{3}\eta_0\!\left(\frac{I_0\,dl}{\lambda}\right)^{\!2}=40\pi^2|I_0|^2\!\left(\frac{dl}{\lambda}\right)^{\!2}\]
\eq{Radiation resistance}{}
\[R_\text{rad}=\frac{2P_\text{rad}}{|I_0|^2}=80\pi^2\!\left(\frac{dl}{\lambda}\right)^{\!2}\;\Omega\]
\textit{Directivity: $D_0=1.5$ (1.76\,dBi);\enspace HPBW$=90°$.}
\end{ebox}

\vspace{0.3pt}
\shead{amberbg}{amber}{HALF-WAVE DIPOLE ($\ell=\lambda/2$)}
\begin{ebox}
\[F(\theta)=\left[\frac{\cos(\frac{\pi}{2}\cos\theta)}{\sin\theta}\right]^2\]
\[R_\text{rad}=73\;\Omega,\quad D_0=1.64\;(2.15\;\text{dBi}),\quad\text{HPBW}=78°\]
\[Z_\text{in}=73+j42.5\;\Omega\;\text{(at } f=c/2\ell\text{; slightly shorter for resonance)}\]
\end{ebox}

\columnbreak

%────── COL 2: ANTENNA PARAMETERS & LINK BUDGET ──────
\shead{greenbg}{green}{ANTENNA PARAMETERS}
\begin{ebox}
\eq{Directivity}{dimensionless}
\[D_0=\frac{4\pi U_\text{max}}{P_\text{rad}}\]
\eq{Radiation efficiency \& gain}{}
\[e_{cd}=\frac{P_\text{rad}}{P_\text{rad}+P_\text{loss}}\in[0,1],\qquad G_0=e_{cd}\,D_0\]
\eq{Effective aperture}{$\text{m}^2$}
\[A_\text{eff}=\frac{\lambda^2 G_0}{4\pi}\]
\eq{Beam solid angle}{}
\[\Omega_A=\iint_\Omega F\,d\Omega=\frac{4\pi}{D_0}\;\text{(sr)}\]
\eq{HPBW (half-power beamwidth)}{angle between $F=0.5$ pts}
\[D_0\approx\frac{4\pi}{\theta_{1\text{HP}}\,\theta_{2\text{HP}}}\;\text{(approx, narrow beam)}\]
\eq{In dB}{$G_\text{dBi}=10\log_{10}(G_0)$; isotropic ref}
\[G_\text{dBi}=10\log_{10}(G_0),\qquad 0\;\text{dBi}\Rightarrow G_0=1\]
\end{ebox}

\vspace{0.3pt}
\shead{greenbg}{green}{FRIIS TRANSMISSION}
\begin{ebox}
\textit{Free space, polarization matched, separation $R$:}
\[\frac{P_r}{P_t}=G_t\,G_r\!\left(\frac{\lambda}{4\pi R}\right)^{\!2}\]
\eq{In dB form}{}
\[P_r[\text{dBW}]=P_t[\text{dBW}]+G_t[\text{dB}]+G_r[\text{dB}]-\text{FSPL}[\text{dB}]\]
\[\text{FSPL}=20\log_{10}\!\left(\frac{4\pi R}{\lambda}\right)\;\text{dB}\]
\textit{Equiv.: $P_r=S_i\,A_\text{eff,r}$ where $S_i=P_t G_t/(4\pi R^2)$.}
\end{ebox}

\vspace{0.3pt}
\shead{greenbg}{green}{RADAR EQUATION}
\begin{ebox}
\eq{Monostatic radar}{$G_t=G_r=G$, range $R$}
\[\frac{P_r}{P_t}=\frac{G^2\lambda^2\,\sigma_{cs}}{(4\pi)^3 R^4}\]
\eq{Maximum range}{}
\[R_\text{max}=\left[\frac{P_t G^2\lambda^2\sigma_{cs}}{(4\pi)^3 P_{r,\text{min}}}\right]^{1/4}\]
\textit{$\sigma_{cs}$=radar cross section (m$^2$); $P_{r,\text{min}}$=minimum detectable power.}\\[0.5pt]
\textit{Power falls as $1/R^4$: doubling range requires $\times16$ transmit power.}
\end{ebox}

\columnbreak

%────── COL 3: RECTANGULAR WAVEGUIDE ──────
\shead{bluebg}{blue}{RECTANGULAR WAVEGUIDE ($a\times b$, $a>b$)}
\begin{ebox}
\textit{Propagation in $+\hat{z}$. Modes: TE$_{mn}$ ($E_z=0$) or TM$_{mn}$ ($H_z=0$).}\\[0.5pt]
\eq{Cutoff wavenumber \& frequency}{}
\[k_{c,mn}=\pi\sqrt{\!\left(\frac{m}{a}\right)^{\!2}+\!\left(\frac{n}{b}\right)^{\!2}},\quad f_{c,mn}=\frac{c}{2}\sqrt{\!\left(\frac{m}{a}\right)^{\!2}+\!\left(\frac{n}{b}\right)^{\!2}}\]
\textit{TE$_{mn}$: $m,n=0,1,2,\ldots$ (not both 0). TM$_{mn}$: $m,n\geq1$.}\\[0.5pt]
\textit{Dominant mode TE$_{10}$ ($m\!=\!1,n\!=\!0$): $f_c=c/(2a)$ (lowest).}\\[0.5pt]
\eq{Single-mode range}{only TE$_{10}$ propagates}
\[\frac{c}{2a}<f<\frac{c}{a}\quad\text{(i.e. }f_{c,10}<f<f_{c,20}\text{)}\]
\end{ebox}

\vspace{0.3pt}
\shead{bluebg}{blue}{WAVEGUIDE WAVE PARAMETERS (f $>$ f$_c$)}
\begin{ebox}
\eq{Propagation constant}{}
\[\beta=\sqrt{k_0^2-k_c^2}=\frac{\omega}{c}\sqrt{1-\left(\frac{f_c}{f}\right)^{\!2}}\;\text{(real, }f>f_c\text{)}\]
\eq{Phase velocity ($>c$) and group velocity ($<c$)}{}
\[u_p=\frac{\omega}{\beta}=\frac{c}{\sqrt{1-(f_c/f)^2}}>c,\qquad u_g=\frac{c^2}{u_p}=c\sqrt{1-\!\left(\frac{f_c}{f}\right)^{\!2}}<c\]
\[\mathbf{u_p\cdot u_g=c^2}\]
\eq{Guide wavelength}{}
\[\lambda_g=\frac{2\pi}{\beta}=\frac{\lambda_0}{\sqrt{1-(f_c/f)^2}}>\lambda_0\]
\eq{Wave impedances}{}
\[\eta_\text{TE}=\frac{\eta_0}{\sqrt{1-(f_c/f)^2}}>\eta_0,\qquad\eta_\text{TM}=\eta_0\sqrt{1-\!\left(\frac{f_c}{f}\right)^{\!2}}<\eta_0\]
\end{ebox}

\vspace{0.3pt}
\shead{bluebg}{blue}{TE$_{10}$ FIELDS}
\begin{ebox}
\textit{Amplitude $H_0$; $x$-dependence $\sin(\pi x/a)$:}
\[\tilde{H}_z=H_0\cos\!\left(\frac{\pi x}{a}\right)e^{-j\beta z},\quad\tilde{H}_y=0\]
\[\tilde{E}_y=-\frac{j\omega\mu a}{\pi}\,H_0\sin\!\left(\frac{\pi x}{a}\right)e^{-j\beta z}\]
\[\tilde{H}_x=\frac{j\beta a}{\pi}\,H_0\sin\!\left(\frac{\pi x}{a}\right)e^{-j\beta z}\]
\[\tilde{E}_x=\tilde{E}_z=\tilde{H}_y=0\]
\textit{Only $\hat{y}$ component of $\mathbf{E}$; vanishes at $x=0$ and $x=a$.}
\end{ebox}
