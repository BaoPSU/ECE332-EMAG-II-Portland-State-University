\shead{purplebg}{purple}{PLANE WAVE PARAMETERS (LOSSLESS)}
\begin{ebox}
\eq{Angular frequency}{}
\[\omega = 2\pi f \qquad f = \frac{\omega}{2\pi}\]

\eq{Phase velocity}{nonmagnetic: $\mu_r=1$}
\[u_p = \frac{c}{\sqrt{\varepsilon_r}} \qquad c = 3\times10^8\;\text{m/s}\]

\eq{Wavenumber $k$}{rad/m}
\[k = \frac{\omega}{u_p} = \frac{2\pi}{\lambda} = \frac{\omega\sqrt{\varepsilon_r}}{c}\]

\eq{Wavelength \& impedance}{}
\[\lambda = \frac{u_p}{f} = \frac{2\pi}{k} \qquad \eta = \frac{\eta_0}{\sqrt{\varepsilon_r}},\;\eta_0=377\,\Omega\]

\textit{\color{coral}Order: $\omega\!\to\!u_p\!\to\!k\!\to\!\lambda\!\to\!\eta$.}
\end{ebox}

\vspace{1pt}
\shead{purplebg}{purple}{LOSSLESS PLANE WAVE — $\mathbf{E}$ AND $\mathbf{H}$}
\begin{ebox}
\eq{General $\mathbf{E}$}{}
\[\mathbf{E} = E_0\cos(\omega t - k\,\hat{k}\cdot\mathbf{r} + \phi_0)\,\hat{e}\]
\textit{$\hat{k}$ = direction of travel (unit vector), e.g.\ $+y\Rightarrow\hat{k}=\hat{y}$.}\\
\textit{$k$ = wavenumber (rad/m): how fast phase changes in space; $k=\omega\sqrt{\varepsilon_r}/c=2\pi/\lambda$.}\\
\textit{$\phi_0$ = initial phase constant (rad): shift set by given condition at known $t$, position.}\\
\textit{$\hat{e}$ = polarization direction of $\mathbf{E}$ (unit vector), e.g.\ E along $x\Rightarrow\hat{e}=\hat{x}$.}\\
\textit{$\mathbf{r}=x\hat{x}+y\hat{y}+z\hat{z}$, so $\hat{k}\cdot\mathbf{r}$ picks the axis, e.g.\ $\hat{k}=\hat{y}\Rightarrow\hat{k}\cdot\mathbf{r}=y$.}

\eq{$\mathbf{H}$ from $\mathbf{E}$}{$\hat{k}$, $\hat{E}$, $\hat{H}$ form right-hand set}
\[\mathbf{H} = \frac{1}{\eta}\,(\hat{k}\times\mathbf{E})\]

\eq{Find $\phi_0$}{given $E=E_1$ at $t=0$, position $r_0$}
\[E_0\cos(-kr_0+\phi_0)=E_1\;\Rightarrow\;\phi_0=\cos^{-1}\!\!\left(\tfrac{E_1}{E_0}\right)+kr_0\]
\textit{\color{coral}$\cos^{-1}$ gives $[0,\pi]$; take $\phi_0$ in that range.}
\end{ebox}

\shead{purplebg}{purple}{DIRECTION OF PROPAGATION}
\begin{ebox}
\begin{tabular}{@{}ll@{\quad}ll@{}}
\toprule

\midrule

$\omega t-ky$ & $+\hat{y}$ & $\omega t+ky$ & $-\hat{y}$\\

\bottomrule

\textit{Minus before $k$-axis $\Rightarrow$ $+$ direction. Plus $\Rightarrow$ $-$ direction.}
\end{ebox}

\shead{tealbg}{teal}{POLARIZATION --- GENERAL WAVE}
\begin{ebox}
\[\mathbf{E}(z,t)=a_x\cos(\omega t\!-\!kz)\,\hat{x}+a_y\cos(\omega t\!-\!kz+\delta)\,\hat{y}\]

\eq{Tilt angle $\gamma$ (major axis from $\hat{x}$)}{$\gamma\in[0^\circ,180^\circ)$}
\[\tan(2\gamma)=\frac{2a_xa_y\cos\delta}{a_x^2-a_y^2}\]

\eq{Ellipticity angle $\chi$}{$\chi\in[-45^\circ,45^\circ]$}
\[\sin(2\chi)=\frac{2a_xa_y\sin\delta}{a_x^2+a_y^2}\]

\eq{Linear peak amplitude}{only when $\delta=0$ or $\pi$}
\[|\mathbf{E}|_\text{peak}=\sqrt{a_x^2+a_y^2}\]
\end{ebox}

\vspace{1pt}
\shead{amberbg}{amber}{RHC / LHC PHASORS ($+\hat{z}$ PROPAGATION)}
\begin{ebox}
\[\tilde{\mathbf{E}}_R = a_R(\hat{x}-j\hat{y})e^{-jkz},\qquad\tilde{\mathbf{E}}_L = a_L(\hat{x}+j\hat{y})e^{-jkz}\]

\eq{Linear $=$ RHC $+$ LHC}{}
\[a_x e^{-jkz}\hat{x}=\tilde{\mathbf{E}}_R+\tilde{\mathbf{E}}_L\;\Longrightarrow\;\boxed{a_R=a_L=\tfrac{a_x}{2}}\]

\textit{Match components: $\hat{y}$ gives $a_R\!=\!a_L$; $\hat{x}$ gives $a_R\!=\!a_x/2$.}
\end{ebox}

\shead{coralbg}{coral}{LOSSY MEDIA --- CLASSIFICATION}
\begin{ebox}
\eq{Loss tangent}{}
\[\frac{\varepsilon''}{\varepsilon'}=\frac{\sigma}{\omega\varepsilon_r\varepsilon_0}\]

\begin{tabular}{@{}p{0.25\linewidth}p{0.36\linewidth}p{0.3\linewidth}@{}}
\toprule
$\varepsilon''/\varepsilon'$ & Type & Use\\
\midrule
\rowcolor{rowB}$>100$ & Good conductor & GC approx\\
$[10^{-2},100]$ & Quasi-conductor & Exact eqs\\
\rowcolor{rowB}$<10^{-2}$ & Low-loss dielectric & LD approx\\
\bottomrule
\end{tabular}\\[1pt]
\textit{Dry soil ($\varepsilon_r\!=\!2.5$, $\sigma\!=\!10^{-4}$): $\varepsilon''/\varepsilon'=(7.193\!\times\!10^5)/f$.}
\end{ebox}

\vspace{1pt}
\shead{coralbg}{coral}{EXACT FORMULAS (QUASI-CONDUCTOR)}
\begin{ebox}
\eq{Let $X=\sqrt{1+(\sigma/\omega\varepsilon)^2}$, $\;\varepsilon=\varepsilon_r\varepsilon_0$, $\;\mu=\mu_0$}{}

\eq{$\alpha$ (Np/m) and $\beta$ (rad/m)}{}
\[\alpha=\omega\sqrt{\tfrac{\mu\varepsilon}{2}}\sqrt{X-1},\quad\beta=\omega\sqrt{\tfrac{\mu\varepsilon}{2}}\sqrt{X+1}\]

\eq{$\lambda$, $u_p$, $\eta_c$}{}
\[\lambda=\frac{2\pi}{\beta},\quad u_p=\frac{\omega}{\beta},\quad|\eta_c|=\frac{\eta}{X^{1/2}},\quad\theta_\eta=\tfrac{1}{2}\arctan\!\tfrac{\sigma}{\omega\varepsilon}\]
where $\eta=\eta_0/\sqrt{\varepsilon_r}$.
\end{ebox}

\vspace{1pt}
\shead{coralbg}{coral}{GOOD CONDUCTOR ($\sigma/\omega\varepsilon\gg 100$)}
\begin{ebox}
\[\alpha\approx\beta\approx\sqrt{\pi f\mu_0\sigma}\]
\[u_p=\sqrt{4\pi f/\mu_0\sigma},\quad\lambda=2\pi/\beta\]
\[\eta_c\approx(1+j)\tfrac{\alpha}{\sigma}=(1+j)\sqrt{\pi f\mu_0/\sigma}\quad(\angle45^\circ)\]
\end{ebox}

\vspace{1pt}
\shead{coralbg}{coral}{LOW-LOSS DIELECTRIC ($\sigma/\omega\varepsilon\ll 0.01$)}
\begin{ebox}
\[\alpha\approx\tfrac{\sigma\eta}{2},\quad\beta\approx\tfrac{\omega\sqrt{\varepsilon_r}}{c},\quad u_p\approx\tfrac{c}{\sqrt{\varepsilon_r}},\quad\eta_c\approx\eta\;(\angle\approx0^\circ)\]
\end{ebox}

\vspace{1pt}
\shead{bluebg}{blue}{$\tilde{\mathbf{H}}$ PHASOR $\to$ $\tilde{\mathbf{E}}$ IN LOSSY MEDIA}
\begin{ebox}
\eq{Step 1 — read $\alpha$, $\beta$}{}
\[\tilde{\mathbf{H}}\propto e^{-\alpha y}e^{-j\beta y}\;\Rightarrow\;\gamma=\alpha+j\beta\]

\eq{Step 2 — $\eta_c$ (nonmagnetic: $\mu_r\!=\!1$)}{}
\[\eta_c=\frac{j\omega\mu_0}{\gamma}=\frac{\omega\mu_0(\beta+j\alpha)}{\alpha^2+\beta^2}\]

\eq{Step 3 — find $\tilde{\mathbf{E}}$}{}
\[\tilde{\mathbf{E}}=-\eta_c\,(\hat{k}\times\tilde{\mathbf{H}})\]

\eq{Step 4 — time domain}{}
\[Ae^{j\phi}e^{-\alpha y}\xrightarrow{\operatorname{Re}\{\cdot e^{j\omega t}\}}Ae^{-\alpha y}\cos(\omega t-\beta y+\phi)\]
\textit{\color{coral}Convert each component $A\!+\!jB\to|Z|\angle\theta$ first.}
\end{ebox}

\vspace{1pt}
\shead{greenbg}{green}{AVERAGE POWER DENSITY (POYNTING)}
\begin{ebox}
\eq{General}{W/m$^2$}
\[\mathbf{S}_\text{av}=\tfrac{1}{2}\operatorname{Re}\{\tilde{\mathbf{E}}\times\tilde{\mathbf{H}}^*\}\]

\eq{Lossless shortcut}{$|E_0|^2=a_x^2+a_y^2+a_z^2$}
\[\mathbf{S}_\text{av}=\frac{|E_0|^2}{2\eta}\,\hat{k}\]

\textit{Direction = $\hat{k}$. If wave goes in $-\hat{x}$, then $\mathbf{S}_\text{av}$ is a negative $x$-component.}
\end{ebox}
