\shead{purplebg}{purple}{PLANE WAVE PARAMETERS (LOSSLESS)}
\begin{ebox}
\eq{Angular frequency}{}
\[\omega = 2\pi f \qquad f = \frac{\omega}{2\pi}\]

\eq{Phase velocity}{nonmagnetic: $\mu_r=1$}
\[u_p = \frac{c}{\sqrt{\varepsilon_r}} \qquad c = 3\times10^8\;\text{m/s}\]

\eq{Wavenumber $k$}{rad/m}
\[k = \frac{\omega}{u_p} = \frac{2\pi}{\lambda} = \frac{\omega\sqrt{\varepsilon_r}}{c}\]

\eq{Wavelength \& impedance}{}
\[\lambda = \frac{u_p}{f} = \frac{2\pi}{k} \qquad \eta = \frac{\eta_0}{\sqrt{\varepsilon_r}},\;\eta_0=377\,\Omega\]

\textit{\color{coral}Order: $\omega\!\to\!u_p\!\to\!k\!\to\!\lambda\!\to\!\eta$.}
\end{ebox}

\vspace{0.3pt}
\shead{purplebg}{purple}{LOSSLESS PLANE WAVE вЂ” $\mathbf{E}$ AND $\mathbf{H}$}
\begin{ebox}
\eq{General $\mathbf{E}$}{}
\[\mathbf{E} = E_0\cos(\omega t - k\,\hat{k}\cdot\mathbf{r} + \phi_0)\,\hat{e}\]
\textit{$\hat{k}$ = direction of travel (unit vector), e.g.\ $+y\Rightarrow\hat{k}=\hat{y}$.}\\
\textit{$k$ = wavenumber (rad/m): how fast phase changes in space; $k=\omega\sqrt{\varepsilon_r}/c=2\pi/\lambda$.}\\
\textit{$\phi_0$ = initial phase constant (rad): shift set by given condition at known $t$, position.}\\
\textit{$\hat{e}$ = polarization direction of $\mathbf{E}$ (unit vector), e.g.\ E along $x\Rightarrow\hat{e}=\hat{x}$.}\\
\textit{$\mathbf{r}=x\hat{x}+y\hat{y}+z\hat{z}$, so $\hat{k}\cdot\mathbf{r}$ picks the axis, e.g.\ $\hat{k}=\hat{y}\Rightarrow\hat{k}\cdot\mathbf{r}=y$.}

\eq{$\mathbf{H}$ from $\mathbf{E}$}{$\hat{k}$, $\hat{E}$, $\hat{H}$ form right-hand set}
\[\mathbf{H} = \frac{1}{\eta}\,(\hat{k}\times\mathbf{E})\]

\eq{Find $\phi_0$}{given $E=E_1$ at $t=0$, position $r_0$}
\[E_0\cos(-kr_0+\phi_0)=E_1\;\Rightarrow\;\phi_0=\cos^{-1}\!\!\left(\tfrac{E_1}{E_0}\right)+kr_0\]
\textit{\color{coral}$\cos^{-1}$ gives $[0,\pi]$; take $\phi_0$ in that range.}
\end{ebox}

\vspace{0.3pt}
\shead{purplebg}{purple}{DIRECTION OF PROPAGATION}
\begin{ebox}
\makebox[0.92\linewidth][c]{\begin{tabular}{@{}ll@{\quad}ll@{}}
\toprule

\midrule

$\omega t-ky$ & $+\hat{y}$ & $\omega t+ky$ & $-\hat{y}$\\

\bottomrule

\textit{Minus before $k$-axis $\Rightarrow$ $+$ direction. Plus $\Rightarrow$ $-$ direction.}
\end{ebox}

\end{ebox}

\vspace{0.3pt}
\shead{graybg}{gray}{READ A WAVE EQUATION вЂ” QUICK STEPS}
\begin{ebox}
\textbf{E.g.}\;$\mathbf{E}=20\cos(\omega t - kz)\,\hat{y}$\\[1pt]
\begin{tabular}{@{}cl@{}}
1. & Phase arg $\omega t - k\mathbf{z}$ $\Rightarrow$ $\hat{k}=+\hat{z}$ \textit{(table above)}\\
2. & Unit vec after $\cos(\cdots)$ $=\hat{y}$ $=$ pol.\ dir.\ $\hat{e}$\\
3. & $\eta=\eta_0/\sqrt{\varepsilon_r}$, where $\eta_0=377\,\Omega$\\
4. & $\mathbf{H}=\tfrac{1}{\eta}(\hat{k}\times\mathbf{E})$; look up $\hat{k}\times\hat{e}$ above\\
\end{tabular}\\[1pt]
\textit{\color{coral}$\phi_0$ only needed when an initial condition is given (see box above).}
\end{ebox}

\shead{tealbg}{teal}{POLARIZATION --- GENERAL WAVE}
\begin{ebox}
\[\mathbf{E}(z,t)=a_x\cos(\omega t\!-\!kz)\,\hat{x}+a_y\cos(\omega t\!-\!kz+\delta)\,\hat{y}\quad\text{(V/m)}\]

\eq{Tilt angle $\psi$ (major axis from $\hat{x}$)}{$\psi\in[0,\pi)$ rad}
\[\tan(2\psi)=\frac{2a_xa_y\cos\delta}{a_x^2-a_y^2}\quad(\psi\text{ in rad})\]

\eq{Ellipticity angle $\chi$}{$\chi\in[-\pi/4,\,\pi/4]$ rad}
\[\sin(2\chi)=\frac{2a_xa_y\sin\delta}{a_x^2+a_y^2}\quad(\chi\text{ in rad})\]

\eq{Peak amplitude $|\mathbf{E}|_\text{peak}$}{V/m, linear pol.\ only}
\[|\mathbf{E}|_\text{peak}=\sqrt{a_x^2+a_y^2}\;\text{(V/m)}\]
\end{ebox}

\shead{coralbg}{coral}{LOSSY MEDIA --- CLASSIFICATION}
\begin{ebox}
\eq{Loss tangent}{}
\[\frac{\varepsilon''}{\varepsilon'}=\frac{\sigma}{\omega\varepsilon_r\varepsilon_0}\]

{\centering\begin{tabular}{@{}p{0.25\linewidth}p{0.36\linewidth}p{0.3\linewidth}@{}}
\toprule
$\varepsilon''/\varepsilon'$ & Type & Use\\
\midrule
\rowcolor{rowB}$>100$ & Good conductor & GC approx\\
$0.01$ to $100$ & Quasi-conductor & Exact eqs\\
\rowcolor{rowB}$<0.01$ & Low-loss dielectric & LD approx\\
\bottomrule
\end{tabular}\par}
\textit{Non-magnetic ($\mu_r\!=\!1$): $\mu\!=\!\mu_0$.\quad $\varepsilon'\!=\!\varepsilon_r\varepsilon_0$ ($\varepsilon_r\!=\!1$: vacuum/air).}
\end{ebox}

\vspace{0.3pt}
\shead{coralbg}{coral}{EXACT FORMULAS (QUASI-CONDUCTOR)}
\begin{ebox}
\eq{Let $X=\sqrt{1+(\sigma/\omega\varepsilon)^2}=\sqrt{1+(\varepsilon''/\varepsilon')^2}$, $\;\varepsilon''=\sigma/\omega$, $\;\varepsilon=\varepsilon_r\varepsilon_0$, $\;\mu=\mu_0$}{}

\eq{$\alpha$ and $\beta$}{}
\[\alpha=\omega\sqrt{\tfrac{\mu\varepsilon}{2}}\sqrt{X-1}\;\text{(Np/m)},\quad\beta=\omega\sqrt{\tfrac{\mu\varepsilon}{2}}\sqrt{X+1}\;\text{(rad/m)}\]

\[\lambda=\frac{2\pi}{\beta}\;\text{(m)},\quad u_p=\frac{\omega}{\beta}\;\text{(m/s)}\]
\[\eta_c=\frac{\eta}{X^{1/2}}\angle\theta_\eta\;(\Omega),\quad\theta_\eta=\tfrac{1}{2}\arctan\!\tfrac{\sigma}{\omega\varepsilon}\;\text{(deg)}\]
where $\eta=\eta_0/\sqrt{\varepsilon_r}$.
\end{ebox}

\vspace{0.3pt}
\shead{coralbg}{coral}{GOOD CONDUCTOR ($\sigma/\omega\varepsilon\gg 100$)}
\begin{ebox}
\[\alpha\approx\omega\sqrt{\tfrac{\mu\varepsilon''}{2}}=\omega\sqrt{\tfrac{\mu\sigma}{2\omega}}=\sqrt{\pi f\mu\sigma}\quad\text{(Np/m)}\]
\[\beta\approx\alpha\approx\sqrt{\pi f\mu\sigma}\quad\text{(rad/m)}\]
\[u_p=\omega/\beta\;\text{(m/s)},\quad\lambda=2\pi/\beta\;\text{(m)}\]
\[\eta_c\approx\sqrt{j\tfrac{\mu}{\varepsilon''}}=(1+j)\sqrt{\tfrac{\pi f\mu}{\sigma}}\approx(1+j)\tfrac{\alpha}{\sigma}\quad(\Omega)\]
\end{ebox}

\vspace{0.3pt}
\shead{coralbg}{coral}{LOW-LOSS DIELECTRIC ($\sigma/\omega\varepsilon\ll 0.01$)}
\begin{ebox}
\[\alpha\approx\tfrac{\omega\varepsilon''}{2}\sqrt{\tfrac{\mu}{\varepsilon'}}=\tfrac{\sigma}{2}\sqrt{\tfrac{\mu}{\varepsilon'}}=\tfrac{\sigma\eta}{2}\;\text{(Np/m)}\]
\[\beta\approx\omega\sqrt{\mu\varepsilon'}=\tfrac{\omega\sqrt{\varepsilon_r}}{c}\;\text{(rad/m)},\quad u_p\approx\tfrac{c}{\sqrt{\varepsilon_r}}\;\text{(m/s)}\]
\[\eta_c\approx\sqrt{\tfrac{\mu}{\varepsilon'}}=\eta\;(\Omega,\;\angle\approx0^\circ)\]
\end{ebox}

\vspace{0.3pt}
\shead{bluebg}{blue}{$\tilde{\mathbf{H}}$ PHASOR $\to$ $\tilde{\mathbf{E}}$ IN LOSSY MEDIA}
\begin{ebox}
\eq{Step 1 вЂ” read $\alpha$, $\beta$}{}
\[\tilde{\mathbf{H}}\propto e^{-\alpha y}e^{-j\beta y}\;\Rightarrow\;\gamma=\alpha+j\beta\]

\eq{Step 2 вЂ” $\eta_c$ (nonmagnetic: $\mu_r\!=\!1$)}{}
\[\eta_c=\frac{j\omega\mu_0}{\gamma}=\frac{\omega\mu_0(\beta+j\alpha)}{\alpha^2+\beta^2}\]

\eq{Step 3 вЂ” find $\tilde{\mathbf{E}}$}{}
\[\tilde{\mathbf{E}}=-\eta_c\,(\hat{k}\times\tilde{\mathbf{H}})\]

\eq{Step 4 вЂ” time domain}{}
\[Ae^{j\phi}e^{-\alpha y}\xrightarrow{\operatorname{Re}\{\cdot e^{j\omega t}\}}Ae^{-\alpha y}\cos(\omega t-\beta y+\phi)\]
\textit{\color{coral}Convert each component $A\!+\!jB\to|Z|\angle\theta$ first.}
\end{ebox}

\vspace{0.3pt}
\shead{greenbg}{green}{AVERAGE POWER DENSITY (POYNTING)}
\begin{ebox}
\eq{General}{W/m$^2$}
\[\mathbf{S}_\text{av}=\tfrac{1}{2}\operatorname{Re}\{\tilde{\mathbf{E}}\times\tilde{\mathbf{H}}^*\}\]

\eq{Lossless shortcut}{$|E_0|^2=a_x^2+a_y^2+a_z^2$}
\[\mathbf{S}_\text{av}=\frac{|E_0|^2}{2\eta}\,\hat{k}\]

\textit{Direction = $\hat{k}$. If wave goes in $-\hat{x}$, then $\mathbf{S}_\text{av}$ is a negative $x$-component.}
\end{ebox}
